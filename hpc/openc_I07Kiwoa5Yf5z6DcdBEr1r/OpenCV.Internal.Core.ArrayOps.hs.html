   26 -- | Normalization type
   27 data NormType
   28    = Norm_Inf
   29    | Norm_L1
   30    | Norm_L2
   31    | Norm_L2SQR
   32    | Norm_Hamming
   33    | Norm_Hamming2
   34    | Norm_MinMax
   35      deriving (Show, Eq)
   36 
   37 data NormAbsRel
   38    = NormRelative
   39    | NormAbsolute
   40      deriving (Show, Eq)
   41 
   42 #num NORM_INF
   43 #num NORM_L1
   44 #num NORM_L2
   45 #num NORM_L2SQR
   46 #num NORM_HAMMING
   47 #num NORM_HAMMING2
   48 #num NORM_MINMAX 
   49 
   50 #num NORM_RELATIVE
   51 
   52 marshalNormType :: NormAbsRel -> NormType -> Int32
   53 marshalNormType absRel normType =
   54     case absRel of
   55       NormRelative -> c'normType .|. c'NORM_RELATIVE
   56       NormAbsolute -> c'normType
   57   where
   58     c'normType = case normType of
   59         Norm_Inf      -> c'NORM_INF
   60         Norm_L1       -> c'NORM_L1
   61         Norm_L2       -> c'NORM_L2
   62         Norm_L2SQR    -> c'NORM_L2SQR
   63         Norm_Hamming  -> c'NORM_HAMMING
   64         Norm_Hamming2 -> c'NORM_HAMMING2
   65         Norm_MinMax   -> c'NORM_MINMAX
