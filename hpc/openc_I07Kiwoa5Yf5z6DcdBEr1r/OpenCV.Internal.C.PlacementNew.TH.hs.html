1 {-# language CPP #-}
    2 {-# language QuasiQuotes #-}
    3 {-# language TemplateHaskell #-}
    4 
    5 #ifndef ENABLE_INTERNAL_DOCUMENTATION
    6 {-# OPTIONS_HADDOCK hide #-}
    7 #endif
    8 
    9 module OpenCV.Internal.C.PlacementNew.TH ( mkPlacementNewInstance ) where
   10 
   11 import "base" Data.Monoid ( (<>) )
   12 import qualified "inline-c-cpp" Language.C.Inline.Cpp as C
   13 import "template-haskell" Language.Haskell.TH
   14 import "template-haskell" Language.Haskell.TH.Quote ( quoteExp )
   15 import "this" OpenCV.Internal.C.PlacementNew ( PlacementNew (..) )
   16 
   17 mkPlacementNewInstance :: Name -> DecsQ
   18 mkPlacementNewInstance name =
   19     [d|
   20       instance PlacementNew $(conT (mkName ctypeName)) where
   21           placementNew    = $(placementNewQ)
   22           placementDelete = $(placementDeleteQ)
   23     |]
   24   where
   25     typeName = nameBase name
   26     ctypeName = "C'"  <> typeName
   27 
   28     placementNewQ = do
   29       src <- newName "src"
   30       dst <- newName "dst"
   31       lamE [varP src, varP dst] $
   32         quoteExp C.exp $
   33           "void { new($(" <> typeName <> " * " <> nameBase dst <> ")) cv::" <> typeName <>
   34                    "(*$(" <> typeName <> " * " <> nameBase src <> ")) }"
   35 
   36     placementDeleteQ = do
   37       ptr <- newName "ptr"
   38       lamE [varP ptr] $
   39         quoteExp C.exp $
   40           "void { $(" <> typeName <> " * " <> nameBase ptr <> ")->~" <> typeName <> "() }"
