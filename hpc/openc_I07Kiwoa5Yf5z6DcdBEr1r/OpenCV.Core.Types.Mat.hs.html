   90 -- | Identity matrix
   91 --
   92 -- <http://docs.opencv.org/3.0-last-rst/modules/core/doc/basic_structures.html#mat-eye OpenCV Sphinx doc>
   93 eyeMat
   94     :: ( ToInt32    height
   95        , ToInt32    width
   96        , ToChannels channels
   97        , ToDepth    depth
   98        )
   99     => height   -- ^
  100     -> width    -- ^
  101     -> channels -- ^
  102     -> depth    -- ^
  103     -> Mat (ShapeT (height ::: width ::: Z)) (ChannelsT channels) (DepthT depth)
  104 eyeMat height width channels depth = unsafeCoerceMat $ unsafePerformIO $
  105     fromPtr [CU.exp|Mat * {
  106       new Mat(Mat::eye( $(int32_t c'height)
  107                       , $(int32_t c'width)
  108                       , $(int32_t c'type)
  109                       ))
  110     }|]
  111   where
  112     c'type = marshalFlags depth' channels'
  113 
  114     c'height  = toInt32    height
  115     c'width   = toInt32    width
  116     channels' = toChannels channels
  117     depth'    = toDepth    depth

  119 {- | Extract a sub region from a 2D-matrix (image)
  120 
  121 Example:
  122 
  123 @
  124 matSubRectImg :: Mat ('S ['D, 'D]) ('S 3) ('S Word8)
  125 matSubRectImg = exceptError $
  126     withMatM (h ::: 2 * w ::: Z)
  127              (Proxy :: Proxy 3)
  128              (Proxy :: Proxy Word8)
  129              white $ \imgM -> do
  130       matCopyToM imgM (V2 0 0) birds_512x341 Nothing
  131       matCopyToM imgM (V2 w 0) subImg        Nothing
  132       lift $ rectangle imgM subRect blue 1 LineType_4 0
  133       lift $ rectangle imgM (toRect $ HRect (V2 w 0) (V2 w h) :: Rect2i) blue 1 LineType_4 0
  134   where
  135     subRect = toRect $ HRect (V2 96 131) (V2 90 60)
  136     subImg = exceptError $
  137                resize (ResizeAbs $ toSize $ V2 w h) InterCubic =<<
  138                matSubRect birds_512x341 subRect
  139     [h, w] = miShape $ matInfo birds_512x341
  140 @
  141 
  142 <<doc/generated/examples/matSubRectImg.png matSubRectImg>>
  143 -}
  144 matSubRect
  145     :: Mat ('S [height, width]) channels depth
  146     -> Rect2i
  147     -> CvExcept (Mat ('S ['D, 'D]) channels depth)
  148 matSubRect matIn rect = unsafeWrapException $ do
  149     matOut <- newEmptyMat
  150     handleCvException (pure $ unsafeCoerceMat matOut) $
  151       withPtr matIn  $ \matInPtr  ->
  152       withPtr matOut $ \matOutPtr ->
  153       withPtr rect   $ \rectPtr   ->
  154         [cvExceptU|
  155           *$(Mat * matOutPtr) =
  156             Mat( *$(Mat * matInPtr)
  157                , *$(Rect2i * rectPtr)
  158                );
  159         |]

  161 matCopyTo
  162     :: Mat ('S [dstHeight, dstWidth]) channels depth -- ^
  163     -> V2 Int32 -- ^
  164     -> Mat ('S [srcHeight, srcWidth]) channels depth -- ^
  165     -> Maybe (Mat ('S [srcHeight, srcWidth]) ('S 1) ('S Word8))
  166     -> CvExcept (Mat ('S [dstHeight, dstWidth]) channels depth)
  167 matCopyTo dst topLeft src mbSrcMask = runST $ do
  168     dstM <- thaw dst
  169     eResult <- runExceptT $ matCopyToM dstM topLeft src mbSrcMask
  170     case eResult of
  171       Left err -> pure $ throwE err
  172       Right () -> pure <$> unsafeFreeze dstM

  175 {- | Converts an array to another data type with optional scaling
  176 
  177 <http://docs.opencv.org/3.0-last-rst/modules/core/doc/basic_structures.html?highlight=convertto#mat-convertto OpenCV Sphinx doc>
  178 -}
  179 matConvertTo
  180     :: forall shape channels srcDepth dstDepth
  181      . (ToDepthDS (Proxy dstDepth))
  182     => Maybe Double -- ^ Optional scale factor.
  183     -> Maybe Double -- ^ Optional delta added to the scaled values.
  184     -> Mat shape channels srcDepth
  185     -> CvExcept (Mat shape channels dstDepth)
  186 matConvertTo alpha beta src = unsafeWrapException $ do
  187     dst <- newEmptyMat
  188     handleCvException (pure $ unsafeCoerceMat dst) $
  189       withPtr src $ \srcPtr ->
  190       withPtr dst $ \dstPtr ->
  191         [cvExcept|
  192           $(Mat * srcPtr)->
  193             convertTo( *$(Mat * dstPtr)
  194                      , $(int32_t c'rtype)
  195                      , $(double c'alpha)
  196                      , $(double c'beta)
  197                      );
  198         |]
  199   where
  200     rtype :: Maybe Depth
  201     rtype = dsToMaybe $ toDepthDS (Proxy :: Proxy dstDepth)
  202 
  203     c'rtype = maybe (-1) marshalDepth rtype
  204     c'alpha = maybe 1 realToFrac alpha
  205     c'beta  = maybe 0 realToFrac beta

  212 matCopyToM
  213     :: (PrimMonad m)
  214     => Mut (Mat ('S [dstHeight, dstWidth]) channels depth) (PrimState m) -- ^
  215     -> V2 Int32 -- ^
  216     -> Mat ('S [srcHeight, srcWidth]) channels depth -- ^
  217     -> Maybe (Mat ('S [srcHeight, srcWidth]) ('S 1) ('S Word8))
  218     -> CvExceptT m ()
  219 matCopyToM dstM (V2 x y) src mbSrcMask = ExceptT $
  220     unsafePrimToPrim $ handleCvException (pure ()) $
  221     withPtr dstM $ \dstPtr ->
  222     withPtr src $ \srcPtr ->
  223     withPtr mbSrcMask $ \srcMaskPtr ->
  224       [cvExcept|
  225         const cv::Mat * const srcPtr = $(const Mat * const srcPtr);
  226         const int32_t x = $(int32_t x);
  227         const int32_t y = $(int32_t y);
  228         cv::Mat * srcMaskPtr = $(Mat * srcMaskPtr);
  229         srcPtr->copyTo( $(Mat * dstPtr)
  230                       ->rowRange(y, y + srcPtr->rows)
  231                        .colRange(x, x + srcPtr->cols)
  232                       , srcMaskPtr
  233                         ? cv::_InputArray(*srcMaskPtr)
  234                         : cv::_InputArray(cv::noArray())
  235                       );
  236       |]
