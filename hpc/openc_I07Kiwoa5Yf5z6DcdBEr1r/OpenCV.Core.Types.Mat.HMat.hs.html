   36 data HMat
   37    = HMat
   38      { hmShape    :: ![Int32]
   39      , hmChannels :: !Int32
   40      , hmElems    :: !HElems
   41      } deriving (Show, Eq)
   42 
   43 data HElems
   44    = HElems_8U       !(VU.Vector Word8)
   45    | HElems_8S       !(VU.Vector Int8)
   46    | HElems_16U      !(VU.Vector Word16)
   47    | HElems_16S      !(VU.Vector Int16)
   48    | HElems_32S      !(VU.Vector Int32)
   49    | HElems_32F      !(VU.Vector Float)
   50    | HElems_64F      !(VU.Vector Double)
   51    | HElems_USRTYPE1 !(V.Vector B.ByteString)
   52      deriving (Show, Eq)
   53 
   54 hElemsDepth :: HElems -> Depth
   55 hElemsDepth = \case
   56     HElems_8U       _v -> Depth_8U
   57     HElems_8S       _v -> Depth_8S
   58     HElems_16U      _v -> Depth_16U
   59     HElems_16S      _v -> Depth_16S
   60     HElems_32S      _v -> Depth_32S
   61     HElems_32F      _v -> Depth_32F
   62     HElems_64F      _v -> Depth_64F
   63     HElems_USRTYPE1 _v -> Depth_USRTYPE1
   64 
   65 hElemsLength :: HElems -> Int
   66 hElemsLength = \case
   67     HElems_8U       v -> VG.length v
   68     HElems_8S       v -> VG.length v
   69     HElems_16U      v -> VG.length v
   70     HElems_16S      v -> VG.length v
   71     HElems_32S      v -> VG.length v
   72     HElems_32F      v -> VG.length v
   73     HElems_64F      v -> VG.length v
   74     HElems_USRTYPE1 v -> VG.length v

   76 matToHMat :: Mat shape channels depth -> HMat
   77 matToHMat mat = unsafePerformIO $ withMatData mat $ \step dataPtr -> do
   78     elems <- copyElems info (map fromIntegral step) dataPtr
   79     pure HMat
   80          { hmShape    = miShape    info
   81          , hmChannels = miChannels info
   82          , hmElems    = elems
   83          }
   84   where
   85     info = matInfo mat
   86 
   87     copyElems
   88         :: MatInfo
   89         -> [Int]     -- ^ step
   90         -> Ptr Word8 -- ^ data
   91         -> IO HElems
   92     copyElems (MatInfo shape depth channels) step dataPtr =
   93         case depth of
   94           Depth_8U  -> HElems_8U  <$> copyToVec
   95           Depth_8S  -> HElems_8S  <$> copyToVec
   96           Depth_16U -> HElems_16U <$> copyToVec
   97           Depth_16S -> HElems_16S <$> copyToVec
   98           Depth_32S -> HElems_32S <$> copyToVec
   99           Depth_32F -> HElems_32F <$> copyToVec
  100           Depth_64F -> HElems_64F <$> copyToVec
  101           Depth_USRTYPE1 -> HElems_USRTYPE1 <$> error "todo"
  102       where
  103         copyToVec :: (Storable a, VU.Unbox a) => IO (VU.Vector a)
  104         copyToVec = do
  105             v <- VUM.unsafeNew $ product0 (map fromIntegral shape) * (fromIntegral channels)
  106             forM_ (zip [0,channels..] $ dimPositions $ map fromIntegral shape) $ \(posIx, pos) -> do
  107                 let elemPtr = matElemAddress dataPtr step pos
  108                 forM_ [0 .. channels - 1] $ \channelIx -> do
  109                   e <- peekElemOff elemPtr $ fromIntegral channelIx
  110                   VUM.unsafeWrite v (fromIntegral $ posIx + channelIx) e
  111             VU.unsafeFreeze v
  112 
  113 hMatToMat :: HMat -> Mat 'D 'D 'D
  114 hMatToMat (HMat shape channels elems) = unsafePerformIO $ do
  115     mat <- exceptErrorIO $ newMat sizes channels depth scalar
  116     withMatData mat copyElems
  117     pure mat
  118   where
  119     sizes = V.fromList shape
  120     depth = hElemsDepth elems
  121 
  122     scalar :: Scalar
  123     scalar = toScalar (zero :: V4 CDouble)
  124 
  125     copyElems :: [CSize] -> Ptr Word8 -> IO ()
  126     copyElems step dataPtr = case elems of
  127         HElems_8U       v -> copyFromVec v
  128         HElems_8S       v -> copyFromVec v
  129         HElems_16U      v -> copyFromVec v
  130         HElems_16S      v -> copyFromVec v
  131         HElems_32S      v -> copyFromVec v
  132         HElems_32F      v -> copyFromVec v
  133         HElems_64F      v -> copyFromVec v
  134         HElems_USRTYPE1 _v -> error "todo"
  135       where
  136         copyFromVec :: (Storable a, VU.Unbox a) => VU.Vector a -> IO ()
  137         copyFromVec v =
  138             forM_ (zip [0, fromIntegral channels ..] $ dimPositions (fromIntegral <$> shape)) $ \(posIx, pos) -> do
  139               let elemPtr = matElemAddress dataPtr (fromIntegral <$> step) pos
  140               forM_ [0 .. channels - 1] $ \channelIx ->
  141                 pokeElemOff elemPtr (fromIntegral channelIx) $ VU.unsafeIndex v (fromIntegral $ posIx + channelIx)
  142 
  143 -- | All possible positions (indexes) for a given shape (list of
  144 -- sizes per dimension).
  145 dimPositions :: (Num a, Enum a) => [a] -> [[a]]
  146 dimPositions shape = sequence $ map (enumFromTo 0) $ map pred shape
  147 
  148 product0 :: (Num a) => [a] -> a
  149 product0 [] = 0
  150 product0 xs = product xs
