    9 module OpenCV.Internal.Core.Types.Matx.TH
   10   ( mkMatxType
   11   ) where
   12 
   13 import "base" Data.List ( intercalate )
   14 import "base" Data.Monoid ( (<>) )
   15 import qualified "inline-c" Language.C.Inline.Unsafe as CU
   16 import "template-haskell" Language.Haskell.TH
   17 import "template-haskell" Language.Haskell.TH.Quote ( quoteExp )
   18 import "this" OpenCV.Internal
   19 import "this" OpenCV.Internal.Core.Types.Matx
   20 import "this" OpenCV.Internal.C.PlacementNew.TH ( mkPlacementNewInstance )
   21 import "this" OpenCV.Internal.C.Types
   22 
   23 mkMatxType
   24     :: String  -- ^ Matx type name, for both Haskell and C
   25     -> Integer -- ^ Row dimension
   26     -> Integer -- ^ Column dimension
   27     -> Name    -- ^ Depth type name in Haskell
   28     -> String  -- ^ Depth type name in C
   29     -> Q [Dec]
   30 mkMatxType mTypeNameStr dimR dimC depthTypeName cDepthTypeStr
   31     | dimR < 1 || dimR > 6 || dimC < 1 || dimC > 6 =
   32         fail $ "mkMatxType: Unsupported dimension: " <> show dimR <> "x" <> show dimC
   33     | otherwise =
   34       fmap concat . sequence $
   35         [ (:[]) <$> matxTySynD
   36         , fromPtrDs
   37         , isMatxOpenCVInstanceDs
   38           -- The largest Matx constructor in C++ takes 16 arguments.
   39           -- TODO (RvD): for larger number of arguments we can use the
   40           -- constructor that initializes from a plain array.
   41         , if dimR * dimC <= 16
   42           then newMatxDs
   43           else pure []
   44         , mkPlacementNewInstance mTypeName
   45         ]
   46   where
   47     mTypeName :: Name
   48     mTypeName = mkName mTypeNameStr
   49 
   50     cMatxTypeStr :: String
   51     cMatxTypeStr = mTypeNameStr
   52 
   53     mTypeQ :: Q Type
   54     mTypeQ = conT mTypeName
   55 
   56     depthTypeQ :: Q Type
   57     depthTypeQ = conT depthTypeName
   58 
   59     dimRTypeQ, dimCTypeQ :: Q Type
   60     dimRTypeQ = litT (numTyLit dimR)
   61     dimCTypeQ = litT (numTyLit dimC)
   62 
   63     matxTySynD :: Q Dec
   64     matxTySynD =
   65         tySynD mTypeName
   66                []
   67                ([t|Matx|] `appT` dimRTypeQ `appT` dimCTypeQ `appT` depthTypeQ)
   68 
   69     fromPtrDs :: Q [Dec]
   70     fromPtrDs =
   71         [d|
   72         instance FromPtr $(mTypeQ) where
   73           fromPtr = objFromPtr Matx $ $(finalizerExpQ)
   74         |]
   75       where
   76         finalizerExpQ :: Q Exp
   77         finalizerExpQ = do
   78           ptr <- newName "ptr"
   79           lamE [varP ptr] $
   80             quoteExp CU.exp $
   81               "void { delete $(" <> cMatxTypeStr <> " * " <> nameBase ptr <> ") }"
   82 
   83     isMatxOpenCVInstanceDs :: Q [Dec]
   84     isMatxOpenCVInstanceDs =
   85         [d|
   86         instance IsMatx (Matx $(dimRTypeQ) $(dimCTypeQ)) $(depthTypeQ) where
   87           toMatx   = id
   88           toMatxIO = pure
   89           fromMatx = id
   90         |]
   91 
   92     newMatxDs :: Q [Dec]
   93     newMatxDs = sequence
   94         [ sigD funName funTypeQ
   95         , withVarNames =<< mapM newName fieldNames
   96         ]
   97       where
   98         -- example: Float -> Float -> Float -> Float -> IO Matx22f
   99         funTypeQ :: Q Type
  100         funTypeQ = foldr (\_fieldName acc -> arrowT `appT` depthTypeQ `appT` acc)
  101                          ([t|IO|] `appT` mTypeQ)
  102                          fieldNames
  103 
  104         funName :: Name
  105         funName = mkName $ "new" <> mTypeNameStr
  106 
  107         fieldNames :: [String]
  108         fieldNames = [fieldName r c | r <- [1..dimR], c <- [1..dimC]]
  109           where
  110             fieldName :: Integer -> Integer -> String
  111             fieldName r c = "f" <> show r <> show c
  112 
  113         withVarNames :: [Name] -> Q Dec
  114         withVarNames varNames = funD funName [funClause]
  115           where
  116             funClause :: Q Clause
  117             funClause = clause (map varP varNames) funBody []
  118 
  119             funBody :: Q Body
  120             funBody = normalB $ appE [e|fromPtr|] $ quoteExp CU.exp $ concat
  121                         [ cMatxTypeStr
  122                         , " * { new cv::Matx<"
  123                         , cDepthTypeStr
  124                         , ", "
  125                         , show dimR
  126                         , ", "
  127                         , show dimC
  128                         , ">"
  129                         , "("
  130                         , intercalate ", " (map fieldQuote varNames)
  131                         , ")"
  132                         , "}"
  133                         ]
  134 
  135             fieldQuote :: Name -> String
  136             fieldQuote n = "$(" <> cDepthTypeStr <> " " <> nameBase n <> ")"
