    4 {- |
    5 
    6 The functions in this section perform various geometrical transformations of 2D
    7 images. They do not change the image content but deform the pixel grid and map
    8 this deformed grid to the destination image. In fact, to avoid sampling
    9 artifacts, the mapping is done in the reverse order, from destination to the
   10 source. That is, for each pixel @(x,y)@ of the destination image, the functions
   11 compute coordinates of the corresponding "donor" pixel in the source image and
   12 copy the pixel value:
   13 
   14 @dst(x,y) = src(fx(x,y), fy(x,y))@
   15 
   16 In case when you specify the forward mapping @\<gx,gy> : src -> dst@, the OpenCV
   17 functions first compute the corresponding inverse mapping @\<fx,fy>:dst->src@
   18 and then use the above formula.
   19 
   20 The actual implementations of the geometrical transformations, from the most
   21 generic remap and to the simplest and the fastest resize, need to solve two main
   22 problems with the above formula:
   23 
   24 * Extrapolation of non-existing pixels.
   25 Similarly to the filtering functions described in the previous section, for some
   26 @(x,y)@, either one of @fx(x,y)@, or @fy(x,y)@, or both of them may fall outside
   27 of the image. In this case, an extrapolation method needs to be used. OpenCV
   28 provides the same selection of extrapolation methods as in the filtering
   29 functions. In addition, it provides the method 'BorderTransparent'. This means
   30 that the corresponding pixels in the destination image will not be modified at
   31 all.
   32 
   33 * Interpolation of pixel values.
   34 Usually @fx(x,y)@ and @fy(x,y)@ are floating-point numbers. This means that
   35 @\<fx,fy>@ can be either an affine or perspective transformation, or radial lens
   36 distortion correction, and so on. So, a pixel value at fractional coordinates
   37 needs to be retrieved. In the simplest case, the coordinates can be just rounded
   38 to the nearest integer coordinates and the corresponding pixel can be used. This
   39 is called a nearest-neighbor interpolation. However, a better result can be
   40 achieved by using more sophisticated interpolation methods , where a polynomial
   41 function is fit into some neighborhood of the computed pixel
   42 @(fx(x,y),fy(x,y))@, and then the value of the polynomial at @(fx(x,y),fy(x,y))@
   43 is taken as the interpolated pixel value. In OpenCV, you can choose between
   44 several interpolation methods. See resize for details.
   45 -}

   88 data ResizeAbsRel
   89    = ResizeAbs Size2i -- ^ Resize to an absolute size.
   90    | ResizeRel (V2 Double)
   91      -- ^ Resize with relative factors for both the width and the height.
   92      deriving Show
   93 
   94 marshalResizeAbsRel
   95     :: ResizeAbsRel
   96     -> (Size2i, CDouble, CDouble)
   97 marshalResizeAbsRel (ResizeAbs s) = (s, 0   , 0   )
   98 marshalResizeAbsRel (ResizeRel f) = (s, c'fx, c'fy)
   99   where
  100     s :: Size2i
  101     s = toSize (zero :: V2 Int32)
  102 
  103     (V2 c'fx c'fy) = realToFrac <$> f

  105 {- | Resizes an image
  106 
  107 To shrink an image, it will generally look best with 'InterArea' interpolation,
  108 whereas to enlarge an image, it will generally look best with 'InterCubic'
  109 (slow) or 'InterLinear' (faster but still looks OK).
  110 
  111 Example:
  112 
  113 @
  114 resizeInterAreaImg :: Mat ('S ['D, 'D]) ('S 3) ('S Word8)
  115 resizeInterAreaImg = exceptError $
  116     withMatM (h ::: w + (w \`div` 2) ::: Z)
  117              (Proxy :: Proxy 3)
  118              (Proxy :: Proxy Word8)
  119              transparent $ \imgM -> do
  120       birds_resized <-
  121         pureExcept $ resize (ResizeRel $ pure 0.5) InterArea birds_768x512
  122       matCopyToM imgM (V2 0 0) birds_768x512 Nothing
  123       matCopyToM imgM (V2 w 0) birds_resized Nothing
  124       lift $ arrowedLine imgM (V2 startX y) (V2 pointX y) red 4 LineType_8 0 0.15
  125   where
  126     [h, w] = miShape $ matInfo birds_768x512
  127     startX = round $ fromIntegral w * (0.95 :: Double)
  128     pointX = round $ fromIntegral w * (1.05 :: Double)
  129     y = h \`div` 4
  130 @
  131 
  132 <<doc/generated/examples/resizeInterAreaImg.png resizeInterAreaImg>>
  133 
  134 <http://docs.opencv.org/3.0-last-rst/modules/imgproc/doc/geometric_transformations.html#resize OpenCV Sphinx doc>
  135 -}
  136 resize
  137     :: ResizeAbsRel
  138     -> InterpolationMethod
  139     -> Mat ('S [height, width]) channels depth
  140     -> CvExcept (Mat ('S ['D, 'D]) channels depth)
  141 resize factor interpolationMethod src = unsafeWrapException $ do
  142     dst <- newEmptyMat
  143     handleCvException (pure $ unsafeCoerceMat dst) $
  144       withPtr src   $ \srcPtr   ->
  145       withPtr dst   $ \dstPtr   ->
  146       withPtr    dsize $ \dsizePtr ->
  147         [cvExcept|
  148           cv::resize
  149           ( *$(Mat * srcPtr)
  150           , *$(Mat * dstPtr)
  151           , *$(Size2i * dsizePtr)
  152           , $(double fx)
  153           , $(double fy)
  154           , $(int32_t c'interpolation)
  155           );
  156         |]
  157   where
  158     (dsize, fx, fy) = marshalResizeAbsRel factor
  159     c'interpolation = marshalInterpolationMethod interpolationMethod

  164 {- | Applies an affine transformation to an image
  165 
  166 Example:
  167 
  168 @
  169 rotateBirds :: Mat (ShapeT [2, 3]) ('S 1) ('S Double)
  170 rotateBirds = getRotationMatrix2D (V2 256 170 :: V2 CFloat) 45 0.75
  171 
  172 warpAffineImg :: Birds_512x341
  173 warpAffineImg = exceptError $
  174     warpAffine birds_512x341 rotateBirds InterArea False False (BorderConstant black)
  175 
  176 warpAffineInvImg :: Birds_512x341
  177 warpAffineInvImg = exceptError $
  178     warpAffine warpAffineImg rotateBirds InterCubic True False (BorderConstant black)
  179 @
  180 
  181 <<doc/generated/birds_512x341.png             original        >>
  182 <<doc/generated/examples/warpAffineImg.png    warpAffineImg   >>
  183 <<doc/generated/examples/warpAffineInvImg.png warpAffineInvImg>>
  184 
  185 <http://docs.opencv.org/3.0-last-rst/modules/imgproc/doc/geometric_transformations.html#warpaffine OpenCV Sphinx doc>
  186 -}
  187 warpAffine
  188     :: Mat ('S [height, width]) channels depth -- ^ Source image.
  189     -> Mat (ShapeT [2, 3]) ('S 1) ('S Double) -- ^ Affine transformation matrix.
  190     -> InterpolationMethod
  191     -> Bool -- ^ Perform the inverse transformation.
  192     -> Bool -- ^ Fill outliers.
  193     -> BorderMode -- ^ Pixel extrapolation method.
  194     -> CvExcept (Mat ('S [height, width]) channels depth) -- ^ Transformed source image.
  195 warpAffine src transform interpolationMethod inverse fillOutliers borderMode =
  196     unsafeWrapException $ do
  197       dst <- newEmptyMat
  198       handleCvException (pure $ unsafeCoerceMat dst) $
  199         withPtr src $ \srcPtr ->
  200         withPtr dst $ \dstPtr ->
  201         withPtr transform   $ \transformPtr ->
  202         withPtr    borderValue $ \borderValuePtr ->
  203           [cvExcept|
  204             Mat * src = $(Mat * srcPtr);
  205             cv::warpAffine
  206               ( *src
  207               , *$(Mat * dstPtr)
  208               , *$(Mat * transformPtr)
  209               , src->size()
  210               , $(int32_t c'interpolationMethod) | $(int32_t c'inverse) | $(int32_t c'fillOutliers)
  211               , $(int32_t c'borderMode)
  212               , *$(Scalar * borderValuePtr)
  213               );
  214           |]
  215   where
  216     c'interpolationMethod = marshalInterpolationMethod interpolationMethod
  217     c'inverse      = if inverse      then c'WARP_INVERSE_MAP   else 0
  218     c'fillOutliers = if fillOutliers then c'WARP_FILL_OUTLIERS else 0
  219     (c'borderMode, borderValue) = marshalBorderMode borderMode

  258 -- | Inverts an affine transformation
  259 --
  260 -- <http://docs.opencv.org/3.0-last-rst/modules/imgproc/doc/geometric_transformations.html#invertaffinetransform OpenCV Sphinx doc>
  261 invertAffineTransform
  262     :: Mat (ShapeT [2, 3]) ('S 1) depth -- ^
  263     -> CvExcept (Mat (ShapeT [2, 3]) ('S 1) depth)
  264 invertAffineTransform matIn = unsafeWrapException $ do
  265     matOut <- newEmptyMat
  266     handleCvException (pure $ unsafeCoerceMat matOut) $
  267       withPtr matIn  $ \matInPtr ->
  268       withPtr matOut $ \matOutPtr ->
  269         [cvExcept|
  270            cv::invertAffineTransform(*$(Mat * matInPtr), *$(Mat * matOutPtr));
  271         |]

  273 {- | Calculates an affine matrix of 2D rotation
  274 
  275 <http://docs.opencv.org/3.0-last-rst/modules/imgproc/doc/geometric_transformations.html#getrotationmatrix2d OpenCV Sphinx doc>
  276 -}
  277 getRotationMatrix2D
  278     :: (IsPoint2 point2 CFloat)
  279     => point2 CFloat -- ^ Center of the rotation in the source image.
  280     -> Double
  281        -- ^ Rotation angle in degrees. Positive values mean counter-clockwise
  282        -- rotation (the coordinate origin is assumed to be the top-left corner).
  283     -> Double -- ^ Isotropic scale factor.
  284     -> Mat (ShapeT [2, 3]) ('S 1) ('S Double) -- ^ The output affine transformation, 2x3 floating-point matrix.
  285 getRotationMatrix2D center angle scale = unsafeCoerceMat $ unsafePerformIO $
  286     withPtr (toPoint center) $ \centerPtr ->
  287       fromPtr
  288       [CU.block| Mat * {
  289         return new cv::Mat
  290         ( cv::getRotationMatrix2D
  291           ( *$(Point2f * centerPtr)
  292           , $(double c'angle)
  293           , $(double c'scale)
  294           )
  295         );
  296       }|]
  297   where
  298     c'angle = realToFrac angle
  299     c'scale = realToFrac scale
