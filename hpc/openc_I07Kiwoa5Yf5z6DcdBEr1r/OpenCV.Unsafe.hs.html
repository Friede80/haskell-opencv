   17 unsafeRead
   18     :: (PrimMonad m, Storable value)
   19     => Mut (Mat shape channels depth) (PrimState m)
   20     -> [Int]
   21     -> m value
   22 unsafeRead matM pos =
   23     unsafePrimToPrim $ withMatData (unMut matM) $ \step dataPtr ->
   24       let elemPtr = matElemAddress dataPtr (fromIntegral <$> step) pos
   25       in peek elemPtr
   26 
   27 unsafeWrite
   28     :: (PrimMonad m, Storable value)
   29     => Mut (Mat shape channels depth) (PrimState m)
   30     -> [Int] -> value -> m ()
   31 unsafeWrite matM pos value =
   32     unsafePrimToPrim $ withMatData (unMut matM) $ \step dataPtr ->
   33       let elemPtr = matElemAddress dataPtr (fromIntegral <$> step) pos
   34       in poke elemPtr value
