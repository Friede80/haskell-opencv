  135 {- | Tests whether a 'Mat' is deserving of its type level attributes
  136 
  137 Checks if the properties encoded in the type of a 'Mat' correspond to
  138 the value level representation. For each property that does not hold
  139 this function will produce an error message. If everything checks out
  140 it will produce an empty list.
  141 
  142 The following properties are checked:
  143 
  144  * Dimensionality
  145  * Size of each dimension
  146  * Number of channels
  147  * Depth (data type of elements)
  148 
  149 If a property is explicitly encoded as statically unknown ('D'ynamic)
  150 it will not be checked.
  151 -}
  152 typeCheckMat
  153     :: forall shape channels depth
  154      . ( ToShapeDS    (Proxy shape)
  155        , ToChannelsDS (Proxy channels)
  156        , ToDepthDS    (Proxy depth)
  157        )
  158     => Mat shape channels depth -- ^ The matrix to be checked.
  159     -> [CoerceMatError] -- ^ Error messages.
  160 typeCheckMat mat =
  161        fromMaybe [] (checkShape <$> dsToMaybe dsExpectedShape)
  162     <> maybeToList (dsToMaybe dsExpectedNumChannels >>= checkNumChannels)
  163     <> maybeToList (dsToMaybe dsExpectedDepth >>= checkDepth)
  164   where
  165     mi = matInfo mat
  166 
  167     dsExpectedShape :: DS [DS Int32]
  168     dsExpectedShape = toShapeDS (Proxy :: Proxy shape)
  169 
  170     dsExpectedNumChannels :: DS Int32
  171     dsExpectedNumChannels = toChannelsDS (Proxy :: Proxy channels)
  172 
  173     dsExpectedDepth :: DS Depth
  174     dsExpectedDepth = toDepthDS (Proxy :: Proxy depth)
  175 
  176     checkShape :: [DS Int32] -> [CoerceMatError]
  177     checkShape expectedShape = maybe checkSizes (:[]) dimCheck
  178       where
  179         dimCheck :: Maybe CoerceMatError
  180         dimCheck | expectedDim == actualDim = Nothing
  181                  | otherwise = Just $ ShapeError $ ExpectationError expectedDim actualDim
  182           where
  183             expectedDim = length expectedShape
  184             actualDim = length (miShape mi)
  185 
  186         checkSizes :: [CoerceMatError]
  187         checkSizes = catMaybes $ zipWith3 checkSize [1..] expectedShape (miShape mi)
  188           where
  189             checkSize :: Int -> DS Int32 -> Int32 -> Maybe CoerceMatError
  190             checkSize dimIx dsExpected actual = dsToMaybe dsExpected >>= \expected ->
  191                 if expected == actual
  192                 then Nothing
  193                 else Just $ SizeError dimIx
  194                           $ fromIntegral
  195                             <$> ExpectationError expected actual
  196 
  197     checkNumChannels :: Int32 -> Maybe CoerceMatError
  198     checkNumChannels expectedNumChannels
  199         | miChannels mi == expectedNumChannels = Nothing
  200         | otherwise = Just $ ChannelError
  201                            $ fromIntegral
  202                              <$> ExpectationError expectedNumChannels (miChannels mi)
  203 
  204     checkDepth :: Depth -> Maybe CoerceMatError
  205     checkDepth expectedDepth
  206         | miDepth mi == expectedDepth = Nothing
  207         | otherwise = Just $ DepthError
  208                            $ ExpectationError expectedDepth (miDepth mi)

  210 -- | Relaxes the type level constraints
  211 --
  212 -- Only identical or looser constraints are allowed. For tighter
  213 -- constraints use 'coerceMat'.
  214 --
  215 -- This allows you to \'forget\' type level guarantees for zero
  216 -- cost. Similar to 'unsafeCoerceMat', but totally safe.
  217 --
  218 -- [Identical] @a@ to @b@ with @a ~ b@
  219 -- [Looser]  @(\''S' a)@ to @\''D'@ or @(\''S' a)@ to @(\''S' b)@ with @'MayRelax' a b@
  220 -- [Tighter] @\''D'@ to @(\''S' a)@
  221 relaxMat
  222     :: ( MayRelax shapeIn    shapeOut
  223        , MayRelax channelsIn channelsOut
  224        , MayRelax depthIn    depthOut
  225        )
  226     => Mat shapeIn  channelsIn  depthIn  -- ^ Original 'Mat'.
  227     -> Mat shapeOut channelsOut depthOut -- ^ 'Mat' with relaxed constraints.
  228 relaxMat = unsafeCoerce
  229 
  230 coerceMat
  231     :: ( ToShapeDS    (Proxy shapeOut)
  232        , ToChannelsDS (Proxy channelsOut)
  233        , ToDepthDS    (Proxy depthOut)
  234        )
  235     => Mat shapeIn channelsIn depthIn -- ^
  236     -> CvExcept (Mat shapeOut channelsOut depthOut)
  237 coerceMat matIn | null errors = pure matOut
  238                 | otherwise   = throwE $ CoerceMatError errors
  239   where
  240     matOut = unsafeCoerceMat matIn
  241     errors = typeCheckMat matOut
  242 
  243 unsafeCoerceMat
  244     :: Mat shapeIn  channelsIn  depthIn
  245     -> Mat shapeOut channelsOut depthOut
  246 unsafeCoerceMat = unsafeCoerce

  248 -- | Similar to 'withPtr' in that it keeps the 'ForeignPtr' alive
  249 -- during the execution of the given action but it doesn't extract the 'Ptr'
  250 -- from the 'ForeignPtr'.
  251 keepMatAliveDuring :: Mat shape channels depth -> IO a -> IO a
  252 keepMatAliveDuring mat m = do
  253     x <- m
  254     touchForeignPtr $ unMat mat
  255     pure x

  260 -- TODO (RvD): what happens if we construct a mat with more than 4 channels?
  261 -- A scalar is just 4 values. What would be the default value of the 5th channel?
  262 newMat
  263     :: ( ToShape    shape
  264        , ToChannels channels
  265        , ToDepth    depth
  266        , ToScalar   scalar
  267        -- , MinLengthDS 2 shape
  268        -- , 1 .<=? channels
  269        -- , channels .<=? 512
  270        -- , 2 <= Length shape
  271        -- , 1 <= channels
  272        -- , channels <= 512
  273        )
  274     => shape -- ^
  275     -> channels
  276     -> depth
  277     -> scalar
  278     -> CvExceptT IO (Mat (ShapeT shape) (ChannelsT channels) (DepthT depth))
  279 newMat shape channels depth defValue = ExceptT $ do
  280     dst <- newEmptyMat
  281     handleCvException (pure $ unsafeCoerceMat dst) $
  282       withVector shape' $ \shapePtr ->
  283       withPtr (toScalar defValue) $ \scalarPtr ->
  284       withPtr dst $ \dstPtr ->
  285         [cvExcept|
  286           *$(Mat * dstPtr) =
  287             Mat( $(int32_t c'ndims)
  288                , $(int32_t * shapePtr)
  289                , $(int32_t c'type)
  290                , *$(Scalar * scalarPtr)
  291                );
  292         |]
  293   where
  294     c'ndims = fromIntegral $ VG.length shape'
  295     c'type  = marshalFlags depth' channels'
  296 
  297     shape'    = toShape shape
  298     channels' = toChannels channels
  299     depth'    = toDepth depth
  300 
  301 -- TODO (BvD): Move to some Utility module.
  302 withVector
  303     :: (VG.Vector v a, Storable a)
  304     => v a -- ^
  305     -> (Ptr a -> IO b)
  306     -> IO b
  307 withVector v f =
  308     allocaArray n $ \ptr ->
  309       let go !ix
  310               | ix < n = do
  311                   pokeElemOff ptr ix (VG.unsafeIndex v ix)
  312                   go (ix+1)
  313               | otherwise = f ptr
  314       in go 0
  315   where
  316     n = VG.length v

  318 withMatData
  319     :: Mat shape channels depth -- ^
  320     -> ([CSize] -> Ptr Word8 -> IO a)
  321     -> IO a
  322 withMatData mat f = withPtr mat $ \matPtr ->
  323     alloca $ \(dimsPtr  :: Ptr Int32      ) ->
  324     alloca $ \(stepPtr2 :: Ptr (Ptr CSize)) ->
  325     alloca $ \(dataPtr2 :: Ptr (Ptr Word8)) -> do
  326       [CU.block|void {
  327         const Mat * const matPtr = $(Mat * matPtr);
  328         *$(int32_t *   const dimsPtr ) = matPtr->dims;
  329         *$(size_t  * * const stepPtr2) = matPtr->step.p;
  330         *$(uint8_t * * const dataPtr2) = matPtr->data;
  331       }|]
  332       dims    <- peek dimsPtr
  333       stepPtr <- peek stepPtr2
  334       dataPtr <- peek dataPtr2
  335       step    <- peekArray (fromIntegral dims) stepPtr
  336       f step dataPtr
  337 
  338 matElemAddress :: Ptr Word8 -> [Int] -> [Int] -> Ptr a
  339 matElemAddress dataPtr step pos = dataPtr `plusPtr` offset
  340     where
  341       offset = sum $ zipWith (*) step pos

  430     :: ( ToShape    shape
  431        , ToChannels channels
  432        , ToDepth    depth

  436     -> channels -- ^
  437     -> depth    -- ^

  459 data MatInfo
  460    = MatInfo
  461      { miShape    :: ![Int32]
  462      , miDepth    :: !Depth
  463      , miChannels :: !Int32
  464      }
  465      deriving (Show, Eq)
  466 
  467 matInfo :: Mat shape channels depth -> MatInfo
  468 matInfo mat = unsafePerformIO $
  469     withPtr mat $ \matPtr ->
  470     alloca $ \(flagsPtr :: Ptr Int32) ->
  471     alloca $ \(dimsPtr  :: Ptr Int32) ->
  472     alloca $ \(sizePtr  :: Ptr (Ptr Int32)) -> do
  473       [CU.block|void {
  474         const Mat * const matPtr = $(Mat * matPtr);
  475         *$(int32_t *   const flagsPtr) = matPtr->flags;
  476         *$(int32_t *   const dimsPtr ) = matPtr->dims;
  477         *$(int32_t * * const sizePtr ) = matPtr->size.p;
  478       }|]
  479       (depth, channels) <- unmarshalFlags <$> peek flagsPtr
  480       dims <- peek dimsPtr
  481       size <- peek sizePtr
  482       shape <- peekArray (fromIntegral dims) size
  483       pure MatInfo
  484            { miShape    = shape
  485            , miDepth    = depth
  486            , miChannels = channels
  487            }

  502 class ToShape a where
  503     toShape :: a -> V.Vector Int32
  504 
  505 -- | identity
  506 instance ToShape (V.Vector Int32) where
  507     toShape = id
  508 
  509 -- | direct conversion to 'V.Vector'
  510 instance ToShape [Int32] where
  511     toShape = V.fromList
  512 
  513 -- | empty 'V.Vector'
  514 instance ToShape (Proxy '[]) where
  515     toShape _proxy = V.empty
  516 
  517 -- | fold over the type level list
  518 instance (ToInt32 (Proxy a), ToShape (Proxy as))
  519       => ToShape (Proxy (a ': as)) where
  520     toShape _proxy =
  521         V.cons
  522           (toInt32 (Proxy :: Proxy a))
  523           (toShape (Proxy :: Proxy as))
  524 
  525 -- | empty 'V.Vector'
  526 instance ToShape Z where
  527     toShape Z = V.empty
  528 
  529 -- | fold over ':::'
  530 instance (ToInt32 a, ToShape as) => ToShape (a ::: as) where
  531     toShape (a ::: as) = V.cons (toInt32 a) (toShape as)
