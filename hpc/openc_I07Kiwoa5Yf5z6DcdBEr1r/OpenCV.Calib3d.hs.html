   39 data FundamentalMatMethod
   40    = FM_7Point
   41    | FM_8Point
   42    | FM_Ransac !(Maybe Double) !(Maybe Double)
   43    | FM_Lmeds  !(Maybe Double)
   44      deriving (Show, Eq)
   45 
   46 marshalFundamentalMatMethod :: FundamentalMatMethod -> (Int32, CDouble, CDouble)
   47 marshalFundamentalMatMethod = \case
   48     FM_7Point       -> (c'CV_FM_7POINT, 0, 0)
   49     FM_8Point       -> (c'CV_FM_8POINT, 0, 0)
   50     FM_Ransac p1 p2 -> (c'CV_FM_RANSAC, maybe 3 realToFrac p1, maybe 0.99 realToFrac p2)
   51     FM_Lmeds     p2 -> (c'CV_FM_LMEDS, 0, maybe 0.99 realToFrac p2)
   52 
   53 data WhichImage = Image1 | Image2 deriving (Show, Eq)
   54 
   55 marshalWhichImage :: WhichImage -> Int32
   56 marshalWhichImage = \case
   57     Image1 -> 1
   58     Image2 -> 2

   77 {- | Calculates a fundamental matrix from the corresponding points in two images
   78 
   79 The minimum number of points required depends on the 'FundamentalMatMethod'.
   80 
   81  * 'FM_7Point': @N == 7@
   82  * 'FM_8Point': @N >= 8@
   83  * 'FM_Ransac': @N >= 15@
   84  * 'FM_Lmeds': @N >= 8@
   85 
   86 With 7 points the 'FM_7Point' method is used, despite the given method.
   87 
   88 With more than 7 points the 'FM_7Point' method will be replaced by the
   89 'FM_8Point' method.
   90 
   91 Between 7 and 15 points the 'FM_Ransac' method will be replaced by the
   92 'FM_Lmeds' method.
   93 
   94 With the 'FM_7Point' method and with 7 points the result can contain up to 3
   95 matrices, resulting in either 3, 6 or 9 rows. This is why the number of
   96 resulting rows in tagged as 'D'ynamic. For all other methods the result always
   97 contains 3 rows.
   98 
   99 <http://docs.opencv.org/3.0-last-rst/modules/calib3d/doc/camera_calibration_and_3d_reconstruction.html#findfundamentalmat OpenCV Sphinx doc>
  100 -}
  101 findFundamentalMat
  102     :: (IsPoint2 point2 CDouble)
  103     => V.Vector (point2 CDouble) -- ^ Points from the first image.
  104     -> V.Vector (point2 CDouble) -- ^ Points from the second image.
  105     -> FundamentalMatMethod
  106     -> CvExcept ( Maybe ( Mat ('S '[ 'D, 'S 3 ]) ('S 1) ('S Double)
  107                         , Mat ('S '[ 'D, 'D   ]) ('S 1) ('S Word8 )
  108                         )
  109                 )
  110 findFundamentalMat pts1 pts2 method = do
  111     (fm, pointMask) <- c'findFundamentalMat
  112     -- If the c++ function can't find a fundamental matrix it will
  113     -- retrun an empty matrix. We check for this case by trying to
  114     -- coerce the result to the desired type.
  115     catchE (Just . (, unsafeCoerceMat pointMask) <$> coerceMat fm)
  116            (\case CoerceMatError _msgs -> pure Nothing
  117                   otherError -> throwE otherError
  118            )
  119   where
  120     c'findFundamentalMat = unsafeWrapException $ do
  121       fm   <- newEmptyMat
  122       pointMask <- newEmptyMat
  123       handleCvException (pure (fm, pointMask)) $
  124         withPtr fm $ \fmPtr ->
  125         withPtr pointMask $ \pointMaskPtr ->
  126         withArrayPtr (V.map toPoint pts1) $ \pts1Ptr ->
  127         withArrayPtr (V.map toPoint pts2) $ \pts2Ptr ->
  128           [cvExcept|
  129             cv::_InputArray pts1 = cv::_InputArray($(Point2d * pts1Ptr), $(int32_t c'numPts1));
  130             cv::_InputArray pts2 = cv::_InputArray($(Point2d * pts2Ptr), $(int32_t c'numPts2));
  131             *$(Mat * fmPtr) =
  132               cv::findFundamentalMat
  133               ( pts1
  134               , pts2
  135               , $(int32_t c'method)
  136               , $(double c'p1)
  137               , $(double c'p2)
  138               , *$(Mat * pointMaskPtr)
  139               );
  140           |]
  141 
  142     c'numPts1 = fromIntegral $ V.length pts1
  143     c'numPts2 = fromIntegral $ V.length pts2
  144     (c'method, c'p1, c'p2) = marshalFundamentalMatMethod method

  146 {- | For points in an image of a stereo pair, computes the corresponding epilines in the other image
  147 
  148 <http://docs.opencv.org/3.0-last-rst/modules/calib3d/doc/camera_calibration_and_3d_reconstruction.html#computecorrespondepilines OpenCV Sphinx doc>
  149 -}
  150 computeCorrespondEpilines
  151     :: (IsPoint2 point2 CDouble)
  152     => V.Vector (point2 CDouble) -- ^ Points.
  153     -> WhichImage -- ^ Image which contains the points.
  154     -> Mat (ShapeT [3, 3]) ('S 1) ('S Double) -- ^ Fundamental matrix.
  155     -> CvExcept (Mat ('S ['D, 'S 1]) ('S 3) ('S Double))
  156 computeCorrespondEpilines points whichImage fm = unsafeWrapException $ do
  157     epilines <- newEmptyMat
  158     handleCvException (pure $ unsafeCoerceMat epilines) $
  159       withArrayPtr (V.map toPoint points) $ \pointsPtr ->
  160       withPtr fm       $ \fmPtr       ->
  161       withPtr epilines $ \epilinesPtr -> do
  162         -- Destroy type information about the pointsPtr. We wan't to generate
  163         -- C++ code that works for any type of point. Specifically Point2f and
  164         -- Point2d.
  165         [cvExcept|
  166           cv::_InputArray points =
  167             cv::_InputArray( $(Point2d * pointsPtr)
  168                            , $(int32_t c'numPoints)
  169                            );
  170           cv::computeCorrespondEpilines
  171           ( points
  172           , $(int32_t c'whichImage)
  173           , *$(Mat * fmPtr)
  174           , *$(Mat * epilinesPtr)
  175           );
  176         |]
  177   where
  178     c'numPoints = fromIntegral $ V.length points
  179     c'whichImage = marshalWhichImage whichImage
