  154 -- | Information about the storage requirements of values in C
  155 --
  156 -- This class assumes that the type @a@ is merely a symbol that corresponds with
  157 -- a type in C.
  158 class CSizeOf a where
  159     -- | Computes the storage requirements (in bytes) of values of
  160     -- type @a@ in C.
  161     cSizeOf :: proxy a -> Int
  162 
  163 instance CSizeOf C'Point2i where cSizeOf _proxy = c'sizeof_Point2i
  164 instance CSizeOf C'Point2f where cSizeOf _proxy = c'sizeof_Point2f
  165 instance CSizeOf C'Point2d where cSizeOf _proxy = c'sizeof_Point2d
  166 instance CSizeOf C'Point3i where cSizeOf _proxy = c'sizeof_Point3i
  167 instance CSizeOf C'Point3f where cSizeOf _proxy = c'sizeof_Point3f
  168 instance CSizeOf C'Point3d where cSizeOf _proxy = c'sizeof_Point3d

  170 instance CSizeOf C'Size2f  where cSizeOf _proxy = c'sizeof_Size2f

  175 -- | Equivalent type in C
  176 --
  177 -- Actually a proxy type in Haskell that stands for the equivalent type in C.
  178 type family C (a :: *) :: *
  179 
  180 type instance C (Maybe a) = C a
  181 
  182 -- | Mutable types have the same C equivalent as their unmutable variants.
  183 type instance C (Mut a s) = C a

  187 -- | Perform an IO action with a pointer to the C equivalent of a value
  188 class WithPtr a where
  189     -- | Perform an action with a temporary pointer to the underlying
  190     -- representation of @a@
  191     --
  192     -- The pointer is not guaranteed to be usuable outside the scope of this
  193     -- function. The same warnings apply as for 'withForeignPtr'.
  194     withPtr :: a -> (Ptr (C a) -> IO b) -> IO b
  195 
  196 -- | 'Nothing' is represented as a 'nullPtr'.
  197 instance (WithPtr a) => WithPtr (Maybe a) where
  198     withPtr Nothing    f = f nullPtr
  199     withPtr (Just obj) f = withPtr obj f
  200 
  201 -- | Mutable types use the same underlying representation as unmutable types.
  202 instance (WithPtr a) => WithPtr (Mut a s) where
  203     withPtr = withPtr . unMut
