   97 -- | Identity matrix
   98 --
   99 -- <http://docs.opencv.org/3.0-last-rst/modules/core/doc/basic_structures.html#mat-eye OpenCV Sphinx doc>
  100 eyeMat
  101     :: ( ToInt32    height
  102        , ToInt32    width
  103        , ToChannels channels
  104        , ToDepth    depth
  105        )
  106     => height   -- ^
  107     -> width    -- ^
  108     -> channels -- ^
  109     -> depth    -- ^
  110     -> Mat (ShapeT (height ::: width ::: Z)) (ChannelsT channels) (DepthT depth)
  111 eyeMat height width channels depth = unsafeCoerceMat $ unsafePerformIO $
  112     fromPtr [CU.exp|Mat * {
  113       new Mat(Mat::eye( $(int32_t c'height)
  114                       , $(int32_t c'width)
  115                       , $(int32_t c'type)
  116                       ))
  117     }|]
  118   where
  119     c'type = marshalFlags depth' channels'
  120 
  121     c'height  = toInt32    height
  122     c'width   = toInt32    width
  123     channels' = toChannels channels
  124     depth'    = toDepth    depth

  126 {- | Extract a sub region from a 2D-matrix (image)
  127 
  128 Example:
  129 
  130 @
  131 matSubRectImg :: Mat ('S ['D, 'D]) ('S 3) ('S Word8)
  132 matSubRectImg = exceptError $
  133     withMatM (h ::: 2 * w ::: Z)
  134              (Proxy :: Proxy 3)
  135              (Proxy :: Proxy Word8)
  136              white $ \imgM -> do
  137       matCopyToM imgM (V2 0 0) birds_512x341 Nothing
  138       matCopyToM imgM (V2 w 0) subImg        Nothing
  139       lift $ rectangle imgM subRect blue 1 LineType_4 0
  140       lift $ rectangle imgM (toRect $ HRect (V2 w 0) (V2 w h) :: Rect2i) blue 1 LineType_4 0
  141   where
  142     subRect = toRect $ HRect (V2 96 131) (V2 90 60)
  143     subImg = exceptError $
  144                resize (ResizeAbs $ toSize $ V2 w h) InterCubic =<<
  145                matSubRect birds_512x341 subRect
  146     [h, w] = miShape $ matInfo birds_512x341
  147 @
  148 
  149 <<doc/generated/examples/matSubRectImg.png matSubRectImg>>
  150 -}
  151 matSubRect
  152     :: Mat ('S [height, width]) channels depth
  153     -> Rect2i
  154     -> CvExcept (Mat ('S ['D, 'D]) channels depth)
  155 matSubRect matIn rect = unsafeWrapException $ do
  156     matOut <- newEmptyMat
  157     handleCvException (pure $ unsafeCoerceMat matOut) $
  158       withPtr matIn  $ \matInPtr  ->
  159       withPtr matOut $ \matOutPtr ->
  160       withPtr rect   $ \rectPtr   ->
  161         [cvExceptU|
  162           *$(Mat * matOutPtr) =
  163             Mat( *$(Mat * matInPtr)
  164                , *$(Rect2i * rectPtr)
  165                );
  166         |]

  168 matCopyTo
  169     :: Mat ('S [dstHeight, dstWidth]) channels depth -- ^
  170     -> V2 Int32 -- ^
  171     -> Mat ('S [srcHeight, srcWidth]) channels depth -- ^
  172     -> Maybe (Mat ('S [srcHeight, srcWidth]) ('S 1) ('S Word8))
  173     -> CvExcept (Mat ('S [dstHeight, dstWidth]) channels depth)
  174 matCopyTo dst topLeft src mbSrcMask = runST $ do
  175     dstM <- thaw dst
  176     eResult <- runExceptT $ matCopyToM dstM topLeft src mbSrcMask
  177     case eResult of
  178       Left err -> pure $ throwE err
  179       Right () -> pure <$> unsafeFreeze dstM

  182 {- | Converts an array to another data type with optional scaling
  183 
  184 <http://docs.opencv.org/3.0-last-rst/modules/core/doc/basic_structures.html?highlight=convertto#mat-convertto OpenCV Sphinx doc>
  185 -}
  186 matConvertTo
  187     :: forall shape channels srcDepth dstDepth
  188      . (ToDepthDS (Proxy dstDepth))
  189     => Maybe Double -- ^ Optional scale factor.
  190     -> Maybe Double -- ^ Optional delta added to the scaled values.
  191     -> Mat shape channels srcDepth
  192     -> CvExcept (Mat shape channels dstDepth)
  193 matConvertTo alpha beta src = unsafeWrapException $ do
  194     dst <- newEmptyMat
  195     handleCvException (pure $ unsafeCoerceMat dst) $
  196       withPtr src $ \srcPtr ->
  197       withPtr dst $ \dstPtr ->
  198         [cvExcept|
  199           $(Mat * srcPtr)->
  200             convertTo( *$(Mat * dstPtr)
  201                      , $(int32_t c'rtype)
  202                      , $(double c'alpha)
  203                      , $(double c'beta)
  204                      );
  205         |]
  206   where
  207     rtype :: Maybe Depth
  208     rtype = dsToMaybe $ toDepthDS (Proxy :: Proxy dstDepth)
  209 
  210     c'rtype = maybe (-1) marshalDepth rtype
  211     c'alpha = maybe 1 realToFrac alpha
  212     c'beta  = maybe 0 realToFrac beta

  214 {- | Create a matrix whose elements are defined by a function.
  215 
  216 Example:
  217 
  218 @
  219 matFromFuncImg
  220   :: forall size. (size ~ 300)
  221   => Mat (ShapeT [size, size]) ('S 4) ('S Word8)
  222 matFromFuncImg = exceptError $
  223     matFromFunc
  224       (Proxy :: Proxy [size, size])
  225       (Proxy :: Proxy 4)
  226       (Proxy :: Proxy Word8)
  227       example
  228   where
  229     example [y, x] 0 = 255 - normDist (V2 x y ^-^ bluePt )
  230     example [y, x] 1 = 255 - normDist (V2 x y ^-^ greenPt)
  231     example [y, x] 2 = 255 - normDist (V2 x y ^-^ redPt  )
  232     example [y, x] 3 =       normDist (V2 x y ^-^ alphaPt)
  233     example _pos _channel = error "impossible"
  234 
  235     normDist :: V2 Int -> Word8
  236     normDist v = floor $ min 255 $ 255 * Linear.norm (fromIntegral \<$> v) / s'
  237 
  238     bluePt  = V2 0 0
  239     greenPt = V2 s s
  240     redPt   = V2 s 0
  241     alphaPt = V2 0 s
  242 
  243     s = fromInteger $ natVal (Proxy :: Proxy size) :: Int
  244     s' = fromIntegral s :: Double
  245 @
  246 
  247 <<doc/generated/examples/matFromFuncImg.png matFromFuncImg>>
  248 -}
  249 matFromFunc
  250     :: forall shape channels depth
  251      . ( ToShape    shape
  252        , ToChannels channels
  253        , ToDepth    depth
  254        , Storable   (StaticDepthT depth)
  255        )
  256     => shape
  257     -> channels
  258     -> depth
  259     -> ([Int] -> Int -> StaticDepthT depth) -- ^
  260     -> CvExcept (Mat (ShapeT shape) (ChannelsT channels) (DepthT depth))
  261 matFromFunc shape channels depth func =
  262     withMatM shape channels depth (0 :: V4 Double) $ \matM ->
  263       forM_ positions $ \pos ->
  264         forM_ [0 .. fromIntegral channels' - 1] $ \channel ->
  265            unsafeWrite matM pos channel $ func pos channel
  266   where
  267     positions :: [[Int]]
  268     positions = dimPositions $ V.toList $ V.map fromIntegral shapeVec
  269 
  270     shapeVec :: V.Vector Int32
  271     shapeVec = toShape shape
  272 
  273     channels' :: Int32
  274     channels' = toChannels channels

  280 matCopyToM
  281     :: (PrimMonad m)
  282     => Mut (Mat ('S [dstHeight, dstWidth]) channels depth) (PrimState m) -- ^
  283     -> V2 Int32 -- ^
  284     -> Mat ('S [srcHeight, srcWidth]) channels depth -- ^
  285     -> Maybe (Mat ('S [srcHeight, srcWidth]) ('S 1) ('S Word8))
  286     -> CvExceptT m ()
  287 matCopyToM dstM (V2 x y) src mbSrcMask = ExceptT $
  288     unsafePrimToPrim $ handleCvException (pure ()) $
  289     withPtr dstM $ \dstPtr ->
  290     withPtr src $ \srcPtr ->
  291     withPtr mbSrcMask $ \srcMaskPtr ->
  292       [cvExcept|
  293         const cv::Mat * const srcPtr = $(const Mat * const srcPtr);
  294         const int32_t x = $(int32_t x);
  295         const int32_t y = $(int32_t y);
  296         cv::Mat * srcMaskPtr = $(Mat * srcMaskPtr);
  297         srcPtr->copyTo( $(Mat * dstPtr)
  298                       ->rowRange(y, y + srcPtr->rows)
  299                        .colRange(x, x + srcPtr->cols)
  300                       , srcMaskPtr
  301                         ? cv::_InputArray(*srcMaskPtr)
  302                         : cv::_InputArray(cv::noArray())
  303                       );
  304       |]
