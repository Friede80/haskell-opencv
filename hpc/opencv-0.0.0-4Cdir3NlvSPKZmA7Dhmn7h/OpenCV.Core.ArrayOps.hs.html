   94 matScalarAdd
   95     :: (ToScalar scalar)
   96     => Mat shape channels depth -- ^
   97     -> scalar
   98     -> Mat shape channels depth
   99 matScalarAdd src x = unsafePerformIO $ do
  100     dst <- newEmptyMat
  101     withPtr (toScalar x) $ \xPtr ->
  102       withPtr dst $ \dstPtr ->
  103         withPtr src $ \srcPtr ->
  104           [C.block| void {
  105             *$(Mat * dstPtr) = *$(Mat * srcPtr) + *$(Scalar * xPtr);
  106           }|]
  107     pure $ unsafeCoerceMat dst
  108 
  109 matScalarMult
  110     :: Mat shape channels depth -- ^
  111     -> Double
  112     -> Mat shape channels depth
  113 matScalarMult src x = unsafePerformIO $ do
  114     dst <- newEmptyMat
  115     withPtr dst $ \dstPtr ->
  116       withPtr src $ \srcPtr ->
  117         [C.block| void {
  118           *$(Mat * dstPtr) = *$(Mat * srcPtr) * $(double c'x);
  119         }|]
  120     pure $ unsafeCoerceMat dst
  121   where
  122     c'x = realToFrac x

  543 {- | Divides a multi-channel array into several single-channel arrays.
  544 
  545 Example:
  546 
  547 @
  548 matSplitImg
  549     :: forall (width    :: Nat)
  550               (width3   :: Nat)
  551               (height   :: Nat)
  552               (channels :: Nat)
  553               (depth    :: *)
  554      . ( Mat (ShapeT [height, width]) ('S channels) ('S depth) ~ Birds_512x341
  555        , width3 ~ ((*) width 3)
  556        )
  557     => Mat (ShapeT [height, width3]) ('S channels) ('S depth)
  558 matSplitImg = exceptError $ do
  559     zeroImg <- mkMat (Proxy :: Proxy [height, width])
  560                      (Proxy :: Proxy 1)
  561                      (Proxy :: Proxy depth)
  562                      black
  563     let blueImg  = matMerge $ V.fromList [channelImgs V.! 0, zeroImg, zeroImg]
  564         greenImg = matMerge $ V.fromList [zeroImg, channelImgs V.! 1, zeroImg]
  565         redImg   = matMerge $ V.fromList [zeroImg, zeroImg, channelImgs V.! 2]
  566 
  567     withMatM (Proxy :: Proxy [height, width3])
  568              (Proxy :: Proxy channels)
  569              (Proxy :: Proxy depth)
  570              white $ \imgM -> do
  571       matCopyToM imgM (V2 (w*0) 0) (unsafeCoerceMat blueImg)  Nothing
  572       matCopyToM imgM (V2 (w*1) 0) (unsafeCoerceMat greenImg) Nothing
  573       matCopyToM imgM (V2 (w*2) 0) (unsafeCoerceMat redImg)   Nothing
  574   where
  575     channelImgs = matSplit birds_512x341
  576 
  577     w :: Int32
  578     w = fromInteger $ natVal (Proxy :: Proxy width)
  579 @
  580 
  581 <<doc/generated/examples/matSplitImg.png matSplitImg>>
  582 
  583 <http://docs.opencv.org/3.0-last-rst/modules/core/doc/operations_on_arrays.html#split OpenCV Sphinx doc>
  584 -}
  585 matSplit
  586     :: Mat shape channels depth -- ^
  587     -> V.Vector (Mat shape ('S 1) depth)
  588 matSplit src = unsafePerformIO $
  589     withPtr src $ \srcPtr ->
  590     allocaArray numChans $ \(splitsArray :: Ptr (Ptr C'Mat)) -> do
  591       [C.block| void {
  592         cv::Mat * srcPtr = $(Mat * srcPtr);
  593         int32_t numChans = $(int32_t c'numChans);
  594         cv::Mat *splits = new cv::Mat[numChans];
  595         cv::split(*srcPtr, splits);
  596         for(int i = 0; i < numChans; i++){
  597           $(Mat * * splitsArray)[i] = new cv::Mat(splits[i]);
  598         }
  599         delete [] splits;
  600       }|]
  601       fmap V.fromList . mapM (fromPtr . pure) =<< peekArray numChans splitsArray
  602   where
  603     numChans = fromIntegral $ miChannels $ matInfo src
  604 
  605     c'numChans :: Int32
  606     c'numChans = fromIntegral numChans

  652     :: NormType

  654        -- ^ Optional operation mask; it must have the same size as the input
  655        -- array, depth 'Depth_8U' and 1 channel.
  656     -> Mat shape channels depth -- ^ Input array.
  657     -> CvExcept Double  -- ^ Calculated norm.
  658 norm normType mbMask src = unsafeWrapException $
  659     withPtr src    $ \srcPtr  ->
  660     withPtr mbMask $ \mskPtr  ->
  661     alloca         $ \normPtr ->
  662     handleCvException (realToFrac <$> peek normPtr) $
  663       [cvExcept|
  664         Mat * mskPtr = $(Mat * mskPtr);
  665         *$(double * normPtr) =
  666           cv::norm( *$(Mat * srcPtr)
  667                   , $(int32_t c'normType)
  668                   , mskPtr ? _InputArray(*mskPtr) : _InputArray(cv::noArray())
  669                   );
  670       |]
  671   where
  672     c'normType = marshalNormType NormAbsolute normType

  674 {- | Calculates an absolute difference norm, or a relative difference norm
  675 
  676 <http://docs.opencv.org/3.0-last-rst/modules/core/doc/operations_on_arrays.html#norm OpenCV Sphinx doc>
  677 -}
  678 normDiff
  679     :: NormAbsRel -- ^ Absolute or relative norm.
  680     -> NormType
  681     -> Maybe (Mat shape ('S 1) ('S Word8))
  682        -- ^ Optional operation mask; it must have the same size as the input
  683        -- array, depth 'Depth_8U' and 1 channel.
  684     -> Mat shape channels depth -- ^ First input array.
  685     -> Mat shape channels depth -- ^ Second input array of the same size and type as the first.
  686     -> CvExcept Double -- ^ Calculated norm.
  687 normDiff absRel normType mbMask src1 src2 = unsafeWrapException $
  688     withPtr src1   $ \src1Ptr ->
  689     withPtr src2   $ \src2Ptr ->
  690     withPtr mbMask $ \mskPtr  ->
  691     alloca         $ \normPtr ->
  692     handleCvException (realToFrac <$> peek normPtr) $
  693       [cvExcept|
  694         Mat * mskPtr = $(Mat * mskPtr);
  695         *$(double * normPtr) =
  696           cv::norm( *$(Mat * src1Ptr)
  697                   , *$(Mat * src2Ptr)
  698                   , $(int32_t c'normType)
  699                   , mskPtr ? _InputArray(*mskPtr) : _InputArray(cv::noArray())
  700                   );
  701       |]
  702   where
  703     c'normType = marshalNormType absRel normType

  705 {- | Normalizes the norm or value range of an array
  706 
  707 <http://docs.opencv.org/3.0-last-rst/modules/core/doc/operations_on_arrays.html#normalize OpenCV Sphinx doc>
  708 -}
  709 normalize
  710     :: forall shape channels srcDepth dstDepth
  711      . (ToDepthDS (Proxy dstDepth))
  712     => Double
  713        -- ^ Norm value to normalize to or the lower range boundary in case of
  714        -- the range normalization.
  715     -> Double
  716        -- ^ Upper range boundary in case of the range normalization; it is not
  717        -- used for the norm normalization.
  718     -> NormType
  719     -> Maybe (Mat shape ('S 1) ('S Word8)) -- ^ Optional operation mask.
  720     -> Mat shape channels srcDepth -- ^ Input array.
  721     -> CvExcept (Mat shape channels dstDepth)
  722 normalize alpha beta normType mbMask src = unsafeWrapException $ do
  723     dst <- newEmptyMat
  724     handleCvException (pure $ unsafeCoerceMat dst) $
  725       withPtr src    $ \srcPtr ->
  726       withPtr dst    $ \dstPtr ->
  727       withPtr mbMask $ \mskPtr ->
  728         [cvExcept|
  729           Mat * mskPtr = $(Mat * mskPtr);
  730           cv::normalize( *$(Mat * srcPtr)
  731                        , *$(Mat * dstPtr)
  732                        , $(double c'alpha)
  733                        , $(double c'beta)
  734                        , $(int32_t c'normType)
  735                        , $(int32_t c'dtype)
  736                        , mskPtr ? _InputArray(*mskPtr) : _InputArray(cv::noArray())
  737                        );
  738         |]
  739   where
  740     c'alpha    = realToFrac alpha
  741     c'beta     = realToFrac beta
  742     c'normType = marshalNormType NormAbsolute normType
  743     c'dtype    = maybe (-1) marshalDepth $ dsToMaybe $ toDepthDS (Proxy :: Proxy dstDepth)

  745 {- | Calculates the sum of array elements
  746 
  747 Example:
  748 
  749 @
  750 matSumImg :: Mat (ShapeT [201, 201]) ('S 3) ('S Word8)
  751 matSumImg = exceptError $

  755       (Proxy :: Proxy Word8)

  757         -- Draw a filled circle. Each pixel has a value of (255,255,255)
  758         lift $ circle imgM (pure radius :: V2 Int32) radius white (-1) LineType_8 0
  759         -- Calculate the sum of all pixels.
  760         scalar <- matSumM imgM
  761         let V4 area _y _z _w = fromScalar scalar :: V4 Double
  762         -- Circle area = pi * radius * radius
  763         let approxPi = area \/ 255 \/ (radius * radius)
  764         lift $ putText imgM
  765                        (T.pack $ show approxPi)
  766                        (V2 40 110 :: V2 Int32)
  767                        (Font FontHersheyDuplex NotSlanted 1)
  768                        blue
  769                        1
  770                        LineType_AA

  781 matSum
  782     :: Mat shape channels depth
  783        -- ^ Input array that must have from 1 to 4 channels.
  784     -> CvExcept Scalar
  785 matSum src = runCvExceptST $ matSumM =<< unsafeThaw src

  787 matSumM

  789     => Mut (Mat shape channels depth) (PrimState m)

  792 matSumM srcM = ExceptT $ unsafePrimToPrim $ do

  794     handleCvException (pure s) $
  795       withPtr srcM $ \srcPtr ->

  797         [cvExcept|

  799         |]
