   48 -- | Create a new cascade classifier. Returns 'Nothing' if the classifier
   49 -- is empty after initialization. This usually means that the file could
   50 -- not be loaded (e.g. it doesn't exist, is corrupt, etc.)
   51 newCascadeClassifier :: FilePath -> IO (Maybe CascadeClassifier)
   52 newCascadeClassifier fp = do
   53   cc <- withCString fp $ \c'fp -> fromPtr
   54     [CU.exp| CascadeClassifier * { new CascadeClassifier(cv::String($(const char * c'fp))) } |]
   55   -- TODO: empty() seems to return bogus numbers when the classifier is not
   56   -- empty, and I'm not sure why. This is also why I'm not using toBool.
   57   empty <- fmap (== 1) (withPtr cc (\ccPtr -> [CU.exp| bool { $(CascadeClassifier * ccPtr)->empty() } |]))
   58   return $ if empty
   59     then Nothing
   60     else Just cc

   62 {- |
   63 Example:
   64 
   65 @
   66 cascadeClassifierArnold
   67   :: forall (width    :: Nat)
   68             (height   :: Nat)
   69             (channels :: Nat)
   70             (depth    :: *  )
   71    . (Mat (ShapeT [height, width]) ('S channels) ('S depth) ~ Arnold_small)
   72   => IO (Mat (ShapeT [height, width]) ('S channels) ('S depth))
   73 cascadeClassifierArnold = do
   74     -- Create two classifiers from data files.
   75     Just ccFrontal <- newCascadeClassifier "data/haarcascade_frontalface_default.xml"
   76     Just ccEyes    <- newCascadeClassifier "data/haarcascade_eye.xml"
   77     -- Detect some features.
   78     let eyes  = ccDetectMultiscale ccEyes    arnoldGray
   79         faces = ccDetectMultiscale ccFrontal arnoldGray
   80     -- Draw the result.
   81     pure $ exceptError $
   82       withMatM (Proxy :: Proxy [height, width])
   83                (Proxy :: Proxy channels)
   84                (Proxy :: Proxy depth)
   85                white $ \imgM -> do
   86         void $ matCopyToM imgM (V2 0 0) arnold_small Nothing
   87         forM_ eyes  $ \eyeRect  -> lift $ rectangle imgM eyeRect  blue  2 LineType_8 0
   88         forM_ faces $ \faceRect -> lift $ rectangle imgM faceRect green 2 LineType_8 0
   89   where
   90     arnoldGray = exceptError $ cvtColor bgr gray arnold_small
   91 
   92     ccDetectMultiscale cc = cascadeClassifierDetectMultiScale cc Nothing Nothing minSize maxSize
   93 
   94     minSize = Nothing :: Maybe (V2 Int32)
   95     maxSize = Nothing :: Maybe (V2 Int32)
   96 @
   97 
   98 <<doc/generated/examples/cascadeClassifierArnold.png cascadeClassifierArnold>>
   99 -}
  100 cascadeClassifierDetectMultiScale
  101   :: (IsSize size Int32)
  102   => CascadeClassifier
  103   -> Maybe Double -- ^ Scale factor, default is 1.1
  104   -> Maybe Int32 -- ^ Min neighbours, default 3
  105   -> Maybe (size Int32) -- ^ Minimum size. Default: no minimum.
  106   -> Maybe (size Int32) -- ^ Maximum size. Default: no maximum.
  107   -> Mat ('S [w, h]) ('S 1) ('S Word8)
  108   -> V.Vector (Rect Int32)
  109 cascadeClassifierDetectMultiScale cc scaleFactor minNeighbours minSize maxSize src = unsafePerformIO $
  110     withPtr cc $ \ccPtr ->
  111     withPtr src $ \srcPtr ->
  112     withPtr c'minSize $ \minSizePtr ->
  113     withPtr c'maxSize $ \maxSizePtr ->
  114     alloca $ \(numRectsPtr :: Ptr Int32) ->
  115     alloca $ \(rectsPtrPtr :: Ptr (Ptr (Ptr (C'Rect Int32)))) -> mask_ $ do
  116       [CU.block| void {
  117         std::vector<cv::Rect> rects;
  118         $(CascadeClassifier * ccPtr)->detectMultiScale(
  119           *$(Mat * srcPtr),
  120           rects,
  121           $(double c'scaleFactor),
  122           $(int32_t c'minNeighbours),
  123           0,
  124           *$(Size2i * minSizePtr),
  125           *$(Size2i * maxSizePtr));
  126         *$(int32_t * numRectsPtr) = rects.size();
  127         cv::Rect * * rectsPtr = new cv::Rect * [rects.size()];
  128         *$(Rect2i * * * rectsPtrPtr) = rectsPtr;
  129         for (std::vector<cv::Rect>::size_type i = 0; i != rects.size(); i++) {
  130           rectsPtr[i] = new cv::Rect(rects[i]);
  131         }
  132       } |]
  133       numRects <- fromIntegral <$> peek numRectsPtr
  134       rectsPtr <- peek rectsPtrPtr
  135       rects :: [Rect Int32] <- peekArray numRects rectsPtr >>= mapM (fromPtr . return)
  136       [CU.block| void { delete [] *$(Rect2i * * * rectsPtrPtr); }|]
  137       return (V.fromList rects)
  138   where
  139     c'scaleFactor = maybe 1.1 realToFrac scaleFactor
  140     c'minNeighbours = maybe 3 fromIntegral minNeighbours
  141     c'minSize = maybe (toSize (V2 0 0)) toSize minSize
  142     c'maxSize = maybe (toSize (V2 0 0)) toSize maxSize
