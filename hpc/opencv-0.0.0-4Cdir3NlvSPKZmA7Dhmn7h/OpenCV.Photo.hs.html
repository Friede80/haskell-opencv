   41 data InpaintingMethod
   42    = InpaintNavierStokes
   43      -- ^ Navier-Stokes based method.
   44    | InpaintTelea
   45      -- ^ Method by Alexandru Telea.
   46      deriving Show
   47 
   48 marshalInpaintingMethod :: InpaintingMethod -> Int32
   49 marshalInpaintingMethod = \case
   50   InpaintNavierStokes -> c'INPAINT_NS
   51   InpaintTelea        -> c'INPAINT_TELEA

   53 {- | Restores the selected region in an image using the region neighborhood.
   54 
   55 Example:
   56 
   57 @
   58 inpaintImg
   59     :: forall h h2 w w2 c d
   60      . ( Mat (ShapeT [h, w]) ('S c) ('S d) ~ Bikes_512x341
   61        , h2 ~ ((*) h 2)
   62        , w2 ~ ((*) w 2)
   63        )
   64     => Mat ('S ['S h2, 'S w2]) ('S c) ('S d)
   65 inpaintImg = exceptError $ do
   66     maskInv <- bitwiseNot mask
   67     maskBgr <- cvtColor gray bgr maskInv
   68     damaged <- bitwiseAnd bikes_512x341 maskBgr
   69     repairedNS <- inpaint 3 InpaintNavierStokes damaged mask
   70     repairedT  <- inpaint 3 InpaintTelea        damaged mask
   71     withMatM
   72       (Proxy :: Proxy [h2, w2])
   73       (Proxy :: Proxy c)
   74       (Proxy :: Proxy d)
   75       black $ \imgM -> do
   76         matCopyToM imgM (V2 0 0) damaged Nothing
   77         matCopyToM imgM (V2 w 0) maskBgr Nothing
   78         matCopyToM imgM (V2 0 h) repairedNS Nothing
   79         matCopyToM imgM (V2 w h) repairedT  Nothing
   80   where
   81     mask = damageMask
   82 
   83     w = fromInteger $ natVal (Proxy :: Proxy w)
   84     h = fromInteger $ natVal (Proxy :: Proxy h)
   85 @
   86 
   87 <<doc/generated/examples/inpaintImg.png inpaintImg>>
   88 -}
   89 inpaint
   90    :: (channels `In` [1, 3])
   91    => Double
   92       -- ^ inpaintRadius - Radius of a circular neighborhood of each
   93       -- point inpainted that is considered by the algorithm.
   94    -> InpaintingMethod
   95    -> Mat ('S [h, w]) ('S channels) ('S Word8) -- ^ Input image.
   96    -> Mat ('S [h, w]) ('S 1) ('S Word8) -- ^ Inpainting mask.
   97    -> CvExcept (Mat ('S [h, w]) ('S channels) ('S Word8)) -- ^ Output image.
   98 inpaint inpaintRadius method src inpaintMask = unsafeWrapException $ do
   99     dst <- newEmptyMat
  100     handleCvException (pure $ unsafeCoerceMat dst) $
  101       withPtr src         $ \srcPtr         ->
  102       withPtr inpaintMask $ \inpaintMaskPtr ->
  103       withPtr dst         $ \dstPtr         ->
  104       [cvExcept|
  105         cv::inpaint( *$(Mat * srcPtr)
  106                    , *$(Mat * inpaintMaskPtr)
  107                    , *$(Mat * dstPtr)
  108                    , $(double c'inpaintRadius)
  109                    , $(int32_t c'method)
  110                    );
  111       |]
  112   where
  113     c'method = marshalInpaintingMethod method
  114     c'inpaintRadius = realToFrac inpaintRadius

  138     w = fromInteger $ natVal (Proxy :: Proxy w)

  144 fastNlMeansDenoisingColored
  145    :: Double -- ^ Parameter regulating filter strength for luminance component.
  146              -- Bigger h value perfectly removes noise but also removes image
  147              -- details, smaller h value preserves details but also preserves
  148              -- some noise
  149    -> Double -- ^ The same as h but for color components. For most images value
  150              -- equals 10 will be enough to remove colored noise and do not
  151              -- distort colors
  152    -> Int32  -- ^ templateWindowSize Size in pixels of the template patch that
  153              -- is used to compute weights.
  154              -- Should be odd. Recommended value 7 pixels
  155    -> Int32  -- ^ searchWindowSize. Size in pixels of the window that is used
  156              -- to compute weighted average for given pixel. Should be odd.
  157              -- Affect performance linearly: greater searchWindowsSize
  158              -- - greater denoising time. Recommended value 21 pixels
  159    -> Mat ('S [h, w]) ('S 3) ('S Word8) -- ^ Input image 8-bit 3-channel image.
  160    -> CvExcept (Mat ('S [h, w]) ('S 3) ('S Word8))
  161              -- ^ Output image same size and type as input.
  162 fastNlMeansDenoisingColored h hColor templateWindowSize searchWindowSize src =
  163   unsafeWrapException $ do
  164     dst <- newEmptyMat
  165     handleCvException (pure $ unsafeCoerceMat dst) $
  166       withPtr src         $ \srcPtr         ->
  167       withPtr dst         $ \dstPtr         ->
  168       [cvExcept|
  169         cv::fastNlMeansDenoisingColored( *$(Mat * srcPtr)
  170                                        , *$(Mat * dstPtr)
  171                                        , $(double c'h)
  172                                        , $(double c'hColor)
  173                                        , $(int32_t templateWindowSize)
  174                                        , $(int32_t searchWindowSize)
  175                                        );
  176       |]
  177   where
  178     c'h = realToFrac h
  179     c'hColor = realToFrac hColor

  181 {- | Perform fastNlMeansDenoisingColoredMulti function for colored images.
  182      Denoising is not pre channel but in a different colour space.
  183      This wrapper differs from the original OpenCV version by using all input
  184      images and denoising the middle one. The original version would allow
  185      to have some arbitrary length vector and slide window over it. As we have
  186      to copy the haskell vector before we can use it as `std::vector` on the cpp
  187      side it is easier to trim the vector before sending and use all frames.
  188 
  189 Example:
  190 
  191 @
  192 fastNlMeansDenoisingColoredMultiImg
  193     :: forall h w w2 c d
  194      . ( Mat (ShapeT [h, w]) ('S c) ('S d) ~ Lenna_512x512
  195        , w2 ~ ((*) w 2)
  196        )
  197     => Mat ('S ['S h, 'S w2]) ('S c) ('S d)
  198 fastNlMeansDenoisingColoredMultiImg = exceptError $ do
  199     denoised <- fastNlMeansDenoisingColoredMulti 3 10 7 21 (V.singleton lenna_512x512)
  200     withMatM
  201       (Proxy :: Proxy [h, w2])
  202       (Proxy :: Proxy c)
  203       (Proxy :: Proxy d)
  204       black $ \imgM -> do
  205         matCopyToM imgM (V2 0 0) lenna_512x512 Nothing
  206         matCopyToM imgM (V2 w 0) denoised Nothing
  207   where
  208     w = fromInteger $ natVal (Proxy :: Proxy w)
  209 @

  211 <<doc/generated/examples/fastNlMeansDenoisingColoredMultiImg.png fastNlMeansDenoisingColoredMultiImg>>

  214 fastNlMeansDenoisingColoredMulti
  215    :: Double -- ^ Parameter regulating filter strength for luminance component.
  216              -- Bigger h value perfectly removes noise but also removes image
  217              -- details, smaller h value preserves details but also preserves
  218              -- some noise
  219    -> Double -- ^ The same as h but for color components. For most images value
  220              -- equals 10 will be enough to remove colored noise and do not
  221              -- distort colors
  222    -> Int32  -- ^ templateWindowSize Size in pixels of the template patch that
  223              -- is used to compute weights. Should be odd.
  224              -- Recommended value 7 pixels
  225    -> Int32  -- ^ searchWindowSize. Size in pixels of the window that is used to
  226              -- compute weighted average for given pixel. Should be odd.
  227              -- Affect performance linearly: greater searchWindowsSize -
  228              -- greater denoising time. Recommended value 21 pixels
  229    -> V.Vector (Mat ('S [h, w]) ('S 3) ('S Word8))
  230              -- ^ Vector of odd number of input 8-bit 3-channel images.
  231    -> CvExcept (Mat ('S [h, w]) ('S 3) ('S Word8))
  232              -- ^ Output image same size and type as input.
  233 
  234 fastNlMeansDenoisingColoredMulti h hColor templateWindowSize searchWindowSize srcVec =
  235   unsafeWrapException $ do
  236     dst <- newEmptyMat
  237     handleCvException (pure $ unsafeCoerceMat dst) $
  238       withArrayPtr srcVec $ \srcVecPtr      ->
  239       withPtr dst         $ \dstPtr         ->
  240       [cvExcept|
  241         std::vector<Mat> buffer( $(Mat * srcVecPtr)
  242                     , $(Mat * srcVecPtr) + $(int32_t c'temporalWindowSize) );
  243         cv::fastNlMeansDenoisingColoredMulti( buffer
  244                                             , *$(Mat * dstPtr)
  245                                             , $(int32_t c'imgToDenoiseIndex)
  246                                             , $(int32_t c'temporalWindowSize)
  247                                             , $(double c'h)
  248                                             , $(double c'hColor)
  249                                             , $(int32_t templateWindowSize)
  250                                             , $(int32_t searchWindowSize)
  251                                             );
  252       |]
  253   where
  254     c'h = realToFrac h
  255     c'hColor = realToFrac hColor
  256     c'srcVecLength = fromIntegral $ V.length srcVec
  257     -- if it is not odd we drop the last image
  258     c'temporalWindowSize
  259         | c'srcVecLength `mod` 2 == 1 = c'srcVecLength
  260         | otherwise                   = c'srcVecLength - 1
  261     c'imgToDenoiseIndex = (c'temporalWindowSize - 1) `div` 2

  291 denoise_TVL1
  292    :: Double -- ^ details more is more 2
  293    -> Int32  -- ^ Number of iterations that the algorithm will run
  294    -> V.Vector (Mat ('S [h, w]) ('S 1) ('S Word8))
  295              -- ^ Vector of odd number of input 8-bit 3-channel images.
  296    -> CvExcept (Mat ('S [h, w]) ('S 1) ('S Word8))
  297              -- ^ Output image same size and type as input.
  298 
  299 denoise_TVL1 lambda niters srcVec = unsafeWrapException $ do
  300     dst <- newEmptyMat
  301     handleCvException (pure $ unsafeCoerceMat dst) $
  302       withArrayPtr srcVec $ \srcVecPtr      ->
  303       withPtr dst         $ \dstPtr         ->
  304       [cvExcept|
  305         std::vector<Mat> buffer( $(Mat * srcVecPtr)
  306                            , $(Mat * srcVecPtr) + $(int32_t c'srcVecLength) );
  307         cv::denoise_TVL1( buffer
  308                         , *$(Mat * dstPtr)
  309                         , $(double c'lambda)
  310                         , $(int32_t niters)
  311                         );
  312       |]
  313   where
  314     c'lambda = realToFrac lambda
  315     c'srcVecLength = fromIntegral $ V.length srcVec
