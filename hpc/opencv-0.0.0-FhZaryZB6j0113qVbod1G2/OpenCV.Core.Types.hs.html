  102 --  RotatedRect
  103 --------------------------------------------------------------------------------

  105 mkRotatedRect
  106     :: ( IsPoint2 point2 CFloat
  107        , IsSize   size   CFloat
  108        )
  109     => point2 CFloat -- ^ Rectangle mass center
  110     -> size   CFloat -- ^ Width and height of the rectangle
  111     -> Float
  112        -- ^ The rotation angle (in degrees). When the angle is 0, 90,
  113        -- 180, 270 etc., the rectangle becomes an up-right rectangle.
  114     -> RotatedRect
  115 mkRotatedRect center size angle =
  116     unsafePerformIO $ newRotatedRect center size (realToFrac angle)
  117 
  118 -- | Rectangle mass center
  119 rotatedRectCenter :: RotatedRect -> Point2f
  120 rotatedRectCenter rotRect = unsafePerformIO $ fromPtr $
  121       withPtr rotRect $ \rotRectPtr ->
  122         [CU.exp| Point2f * { new Point2f($(RotatedRect * rotRectPtr)->center) }|]
  123 
  124 -- | Width and height of the rectangle
  125 rotatedRectSize :: RotatedRect -> Size2f
  126 rotatedRectSize rotRect = unsafePerformIO $ fromPtr $
  127       withPtr rotRect $ \rotRectPtr ->
  128         [CU.exp| Size2f * { new Size2f($(RotatedRect * rotRectPtr)->size) }|]
  129 
  130 -- | The rotation angle (in degrees)
  131 --
  132 -- When the angle is 0, 90, 180, 270 etc., the rectangle becomes an
  133 -- up-right rectangle.
  134 rotatedRectAngle :: RotatedRect -> Float
  135 rotatedRectAngle rotRect = realToFrac $ unsafePerformIO $
  136     withPtr rotRect $ \rotRectPtr ->
  137       [CU.exp| float { $(RotatedRect * rotRectPtr)->angle }|]
  138 
  139 -- | The minimal up-right rectangle containing the rotated rectangle
  140 rotatedRectBoundingRect :: RotatedRect -> Rect2i
  141 rotatedRectBoundingRect rotRect =
  142     unsafePerformIO $ fromPtr $ withPtr rotRect $ \rotRectPtr ->
  143       [CU.exp| Rect2i * { new Rect2i($(RotatedRect * rotRectPtr)->boundingRect()) }|]
  144 
  145 rotatedRectPoints :: RotatedRect -> (Point2f, Point2f, Point2f, Point2f)
  146 rotatedRectPoints rotRect = unsafePerformIO $ do
  147     p1 <- toPointIO (zero :: V2 CFloat)
  148     p2 <- toPointIO (zero :: V2 CFloat)
  149     p3 <- toPointIO (zero :: V2 CFloat)
  150     p4 <- toPointIO (zero :: V2 CFloat)
  151     withPtr rotRect $ \rotRectPtr ->
  152       withPtr p1 $ \p1Ptr ->
  153       withPtr p2 $ \p2Ptr ->
  154       withPtr p3 $ \p3Ptr ->
  155       withPtr p4 $ \p4Ptr ->
  156         [C.block| void {
  157           Point2f vertices[4];
  158           $(RotatedRect * rotRectPtr)->points(vertices);
  159           *$(Point2f * p1Ptr) = vertices[0];
  160           *$(Point2f * p2Ptr) = vertices[1];
  161           *$(Point2f * p3Ptr) = vertices[2];
  162           *$(Point2f * p4Ptr) = vertices[3];
  163         }|]
  164     pure (p1, p2, p3, p4)

  191 -- KeyPoint

  193 
  194 {- | Data structure for salient point detectors
  195 
  196 <http://docs.opencv.org/3.0-last-rst/modules/core/doc/basic_structures.html#keypoint OpenCV Sphinx doc>
  197 -}
  198 newtype KeyPoint = KeyPoint {unKeyPoint :: ForeignPtr C'KeyPoint}
  199 
  200 type instance C KeyPoint = C'KeyPoint
  201 
  202 instance WithPtr KeyPoint where
  203     withPtr = withForeignPtr . unKeyPoint
  204 
  205 instance FromPtr KeyPoint where
  206     fromPtr = objFromPtr KeyPoint $ \ptr ->
  207                 [CU.exp| void { delete $(KeyPoint * ptr) }|]
  208 
  209 instance CSizeOf C'KeyPoint where
  210     cSizeOf _proxy = c'sizeof_KeyPoint
  211 
  212 data KeyPointRec
  213    = KeyPointRec
  214      { kptPoint :: !(V2 Float)
  215        -- ^ Coordinates of the keypoints.
  216      , kptSize :: !Float
  217        -- ^ Diameter of the meaningful keypoint neighborhood.
  218      , kptAngle :: !Float
  219        -- ^ Computed orientation of the keypoint (-1 if not applicable); it's in
  220        -- [0,360) degrees and measured relative to image coordinate system, ie
  221        -- in clockwise.
  222      , kptResponse :: !Float
  223        -- ^ The response by which the most strong keypoints have been
  224        -- selected. Can be used for the further sorting or subsampling.
  225      , kptOctave :: !Int32
  226        -- ^ Octave (pyramid layer) from which the keypoint has been extracted.
  227      , kptClassId :: !Int32
  228        -- ^ Object class (if the keypoints need to be clustered by an object
  229        -- they belong to).
  230      } deriving (Eq, Show)
  231 
  232 newKeyPoint :: KeyPointRec -> IO KeyPoint
  233 newKeyPoint KeyPointRec{..} = fromPtr $
  234     [CU.exp|KeyPoint * {
  235       new cv::KeyPoint
  236           ( cv::Point2f($(float c'x), $(float c'y))
  237           , $(float c'kptSize)
  238           , $(float c'kptAngle)
  239           , $(float c'kptResponse)
  240           , $(int32_t kptOctave)
  241           , $(int32_t kptClassId)
  242           )
  243     }|]
  244   where
  245     V2 c'x c'y = realToFrac <$> kptPoint
  246     c'kptSize     = realToFrac kptSize
  247     c'kptAngle    = realToFrac kptAngle
  248     c'kptResponse = realToFrac kptResponse
  249 
  250 mkKeyPoint :: KeyPointRec -> KeyPoint
  251 mkKeyPoint = unsafePerformIO . newKeyPoint
  252 
  253 keyPointAsRec :: KeyPoint -> KeyPointRec
  254 keyPointAsRec kpt = unsafePerformIO $
  255     withPtr kpt $ \kptPtr ->
  256     alloca $ \xPtr        ->
  257     alloca $ \yPtr        ->
  258     alloca $ \sizePtr     ->
  259     alloca $ \anglePtr    ->
  260     alloca $ \responsePtr ->
  261     alloca $ \octavePtr   ->
  262     alloca $ \classIdPtr  -> do
  263       [CU.block|void {
  264         KeyPoint * kpt = $(KeyPoint * kptPtr);
  265         *$(float   * xPtr       ) = kpt->pt.x    ;
  266         *$(float   * yPtr       ) = kpt->pt.y    ;
  267         *$(float   * sizePtr    ) = kpt->size    ;
  268         *$(float   * anglePtr   ) = kpt->angle   ;
  269         *$(float   * responsePtr) = kpt->response;
  270         *$(int32_t * octavePtr  ) = kpt->octave  ;
  271         *$(int32_t * classIdPtr ) = kpt->class_id;
  272       }|]
  273       KeyPointRec
  274         <$> ( V2 <$> (realToFrac <$> peek xPtr)
  275                  <*> (realToFrac <$> peek yPtr)
  276             )
  277         <*> (realToFrac <$> peek sizePtr    )
  278         <*> (realToFrac <$> peek anglePtr   )
  279         <*> (realToFrac <$> peek responsePtr)
  280         <*> peek octavePtr
  281         <*> peek classIdPtr

  283 --------------------------------------------------------------------------------
  284 -- DMatch
  285 --------------------------------------------------------------------------------
  286 
  287 {- | Class for matching keypoint descriptors: query descriptor index, train
  288 descriptor index, train image index, and distance between descriptors
  289 
  290 <http://docs.opencv.org/3.0-last-rst/modules/core/doc/basic_structures.html#dmatch OpenCV Sphinx Doc>
  291 -}
  292 newtype DMatch = DMatch {unDMatch :: ForeignPtr C'DMatch}
  293 
  294 type instance C DMatch = C'DMatch
  295 
  296 instance WithPtr DMatch where
  297     withPtr = withForeignPtr . unDMatch
  298 
  299 instance FromPtr DMatch where
  300     fromPtr = objFromPtr DMatch $ \ptr ->
  301                 [CU.exp| void { delete $(DMatch * ptr) }|]
  302 
  303 instance CSizeOf C'DMatch where
  304     cSizeOf _proxy = c'sizeof_DMatch
  305 
  306 data DMatchRec
  307    = DMatchRec
  308      { dmatchQueryIdx :: !Int32
  309        -- ^ Query descriptor index.
  310      , dmatchTrainIdx :: !Int32
  311        -- ^ Train descriptor index.
  312      , dmatchImgIdx   :: !Int32
  313        -- ^ Train image index.
  314      , dmatchDistance :: !Float
  315      } deriving (Eq, Show)
  316 
  317 newDMatch :: DMatchRec -> IO DMatch
  318 newDMatch DMatchRec{..} = fromPtr $
  319     [CU.exp|DMatch * {
  320       new cv::DMatch
  321           ( $(int32_t dmatchQueryIdx)
  322           , $(int32_t dmatchTrainIdx)
  323           , $(int32_t dmatchImgIdx)
  324           , $(float c'distance)
  325           )
  326     }|]
  327   where
  328     c'distance = realToFrac dmatchDistance
  329 
  330 mkDMatch :: DMatchRec -> DMatch
  331 mkDMatch = unsafePerformIO . newDMatch
  332 
  333 dmatchAsRec :: DMatch -> DMatchRec
  334 dmatchAsRec dmatch = unsafePerformIO $
  335     withPtr dmatch $ \dmatchPtr ->
  336     alloca $ \queryIdxPtr ->
  337     alloca $ \trainIdxPtr ->
  338     alloca $ \imgIdxPtr ->
  339     alloca $ \distancePtr -> do
  340       [CU.block|void {
  341         DMatch * dmatch = $(DMatch * dmatchPtr);
  342         *$(int32_t * queryIdxPtr) = dmatch->queryIdx;
  343         *$(int32_t * trainIdxPtr) = dmatch->trainIdx;
  344         *$(int32_t * imgIdxPtr  ) = dmatch->imgIdx  ;
  345         *$(float   * distancePtr) = dmatch->distance;
  346       }|]
  347       DMatchRec
  348         <$> peek queryIdxPtr
  349         <*> peek trainIdxPtr
  350         <*> peek imgIdxPtr
  351         <*> (realToFrac <$> peek distancePtr)
