    4 {- |
    5 
    6 The functions in this section perform various geometrical transformations of 2D
    7 images. They do not change the image content but deform the pixel grid and map
    8 this deformed grid to the destination image. In fact, to avoid sampling
    9 artifacts, the mapping is done in the reverse order, from destination to the
   10 source. That is, for each pixel @(x,y)@ of the destination image, the functions
   11 compute coordinates of the corresponding "donor" pixel in the source image and
   12 copy the pixel value:
   13 
   14 @dst(x,y) = src(fx(x,y), fy(x,y))@
   15 
   16 In case when you specify the forward mapping @\<gx,gy> : src -> dst@, the OpenCV
   17 functions first compute the corresponding inverse mapping @\<fx,fy>:dst->src@
   18 and then use the above formula.
   19 
   20 The actual implementations of the geometrical transformations, from the most
   21 generic remap and to the simplest and the fastest resize, need to solve two main
   22 problems with the above formula:
   23 
   24 * Extrapolation of non-existing pixels.
   25 Similarly to the filtering functions described in the previous section, for some
   26 @(x,y)@, either one of @fx(x,y)@, or @fy(x,y)@, or both of them may fall outside
   27 of the image. In this case, an extrapolation method needs to be used. OpenCV
   28 provides the same selection of extrapolation methods as in the filtering
   29 functions. In addition, it provides the method 'BorderTransparent'. This means
   30 that the corresponding pixels in the destination image will not be modified at
   31 all.
   32 
   33 * Interpolation of pixel values.
   34 Usually @fx(x,y)@ and @fy(x,y)@ are floating-point numbers. This means that
   35 @\<fx,fy>@ can be either an affine or perspective transformation, or radial lens
   36 distortion correction, and so on. So, a pixel value at fractional coordinates
   37 needs to be retrieved. In the simplest case, the coordinates can be just rounded
   38 to the nearest integer coordinates and the corresponding pixel can be used. This
   39 is called a nearest-neighbor interpolation. However, a better result can be
   40 achieved by using more sophisticated interpolation methods , where a polynomial
   41 function is fit into some neighborhood of the computed pixel
   42 @(fx(x,y),fy(x,y))@, and then the value of the polynomial at @(fx(x,y),fy(x,y))@
   43 is taken as the interpolated pixel value. In OpenCV, you can choose between
   44 several interpolation methods. See resize for details.
   45 -}

   92 data ResizeAbsRel
   93    = ResizeAbs Size2i -- ^ Resize to an absolute size.
   94    | ResizeRel (V2 Double)
   95      -- ^ Resize with relative factors for both the width and the height.
   96      deriving Show
   97 
   98 marshalResizeAbsRel
   99     :: ResizeAbsRel
  100     -> (Size2i, CDouble, CDouble)
  101 marshalResizeAbsRel (ResizeAbs s) = (s, 0   , 0   )
  102 marshalResizeAbsRel (ResizeRel f) = (s, c'fx, c'fy)
  103   where
  104     s :: Size2i
  105     s = toSize (zero :: V2 Int32)
  106 
  107     (V2 c'fx c'fy) = realToFrac <$> f

  109 {- | Resizes an image
  110 
  111 To shrink an image, it will generally look best with 'InterArea' interpolation,
  112 whereas to enlarge an image, it will generally look best with 'InterCubic'
  113 (slow) or 'InterLinear' (faster but still looks OK).
  114 
  115 Example:
  116 
  117 @
  118 resizeInterAreaImg :: Mat ('S ['D, 'D]) ('S 3) ('S Word8)
  119 resizeInterAreaImg = exceptError $
  120     withMatM (h ::: w + (w \`div` 2) ::: Z)
  121              (Proxy :: Proxy 3)
  122              (Proxy :: Proxy Word8)
  123              transparent $ \imgM -> do
  124       birds_resized <-
  125         pureExcept $ resize (ResizeRel $ pure 0.5) InterArea birds_768x512
  126       matCopyToM imgM (V2 0 0) birds_768x512 Nothing
  127       matCopyToM imgM (V2 w 0) birds_resized Nothing
  128       lift $ arrowedLine imgM (V2 startX y) (V2 pointX y) red 4 LineType_8 0 0.15
  129   where
  130     [h, w] = miShape $ matInfo birds_768x512
  131     startX = round $ fromIntegral w * (0.95 :: Double)
  132     pointX = round $ fromIntegral w * (1.05 :: Double)
  133     y = h \`div` 4
  134 @
  135 
  136 <<doc/generated/examples/resizeInterAreaImg.png resizeInterAreaImg>>
  137 
  138 <http://docs.opencv.org/3.0-last-rst/modules/imgproc/doc/geometric_transformations.html#resize OpenCV Sphinx doc>
  139 -}
  140 resize
  141     :: ResizeAbsRel
  142     -> InterpolationMethod
  143     -> Mat ('S [height, width]) channels depth
  144     -> CvExcept (Mat ('S ['D, 'D]) channels depth)
  145 resize factor interpolationMethod src = unsafeWrapException $ do
  146     dst <- newEmptyMat
  147     handleCvException (pure $ unsafeCoerceMat dst) $
  148       withPtr src   $ \srcPtr   ->
  149       withPtr dst   $ \dstPtr   ->
  150       withPtr dsize $ \dsizePtr ->
  151         [cvExcept|
  152           cv::resize
  153           ( *$(Mat * srcPtr)
  154           , *$(Mat * dstPtr)
  155           , *$(Size2i * dsizePtr)
  156           , $(double fx)
  157           , $(double fy)
  158           , $(int32_t c'interpolation)
  159           );
  160         |]
  161   where
  162     (dsize, fx, fy) = marshalResizeAbsRel factor
  163     c'interpolation = marshalInterpolationMethod interpolationMethod

  191 warpAffine
  192     :: Mat ('S [height, width]) channels depth -- ^ Source image.
  193     -> Mat (ShapeT [2, 3]) ('S 1) ('S Double) -- ^ Affine transformation matrix.
  194     -> InterpolationMethod
  195     -> Bool -- ^ Perform the inverse transformation.
  196     -> Bool -- ^ Fill outliers.
  197     -> BorderMode -- ^ Pixel extrapolation method.
  198     -> CvExcept (Mat ('S [height, width]) channels depth) -- ^ Transformed source image.
  199 warpAffine src transform interpolationMethod inverse fillOutliers borderMode =
  200     unsafeWrapException $ do
  201       dst <- newEmptyMat
  202       handleCvException (pure $ unsafeCoerceMat dst) $
  203         withPtr src $ \srcPtr ->
  204         withPtr dst $ \dstPtr ->
  205         withPtr transform   $ \transformPtr ->
  206         withPtr    borderValue $ \borderValuePtr ->
  207           [cvExcept|
  208             Mat * src = $(Mat * srcPtr);
  209             cv::warpAffine
  210               ( *src
  211               , *$(Mat * dstPtr)
  212               , *$(Mat * transformPtr)
  213               , src->size()
  214               , $(int32_t c'interpolationMethod) | $(int32_t c'inverse) | $(int32_t c'fillOutliers)
  215               , $(int32_t c'borderMode)
  216               , *$(Scalar * borderValuePtr)
  217               );
  218           |]
  219   where
  220     c'interpolationMethod = marshalInterpolationMethod interpolationMethod
  221     c'inverse      = if inverse      then c'WARP_INVERSE_MAP   else 0
  222     c'fillOutliers = if fillOutliers then c'WARP_FILL_OUTLIERS else 0
  223     (c'borderMode, borderValue) = marshalBorderMode borderMode

  277 {- | Calculates a perspective transformation matrix for 2D perspective transform
  278 
  279 <http://docs.opencv.org/3.0-last-rst/modules/imgproc/doc/geometric_transformations.html#getperspectivetransform OpenCV Sphinx doc>
  280 -}
  281 getPerspectiveTransform
  282     :: (IsPoint2 point2 CFloat)
  283     => V.Vector (point2 CFloat) -- ^ Array of 4 floating-point Points representing 4 vertices in source image
  284     -> V.Vector (point2 CFloat) -- ^ Array of 4 floating-point Points representing 4 vertices in destination image
  285     -> Mat (ShapeT [3,3]) ('S 1) ('S Double) -- ^ The output perspective transformation, 3x3 floating-point-matrix.
  286 getPerspectiveTransform srcPts dstPts = unsafeCoerceMat $ unsafePerformIO $
  287     withArrayPtr (V.map toPoint srcPts) $ \srcPtsPtr ->
  288         withArrayPtr (V.map toPoint dstPts) $ \dstPtsPtr ->
  289         fromPtr
  290         [CU.block| Mat * {
  291             return new cv::Mat
  292             ( cv::getPerspectiveTransform($(Point2f * srcPtsPtr), $(Point2f * dstPtsPtr))
  293             );
  294         }|]

  296 {- | Calculates an affine matrix of 2D rotation
  297 
  298 <http://docs.opencv.org/3.0-last-rst/modules/imgproc/doc/geometric_transformations.html#getrotationmatrix2d OpenCV Sphinx doc>
  299 -}
  300 getRotationMatrix2D
  301     :: (IsPoint2 point2 CFloat)
  302     => point2 CFloat -- ^ Center of the rotation in the source image.
  303     -> Double
  304        -- ^ Rotation angle in degrees. Positive values mean counter-clockwise
  305        -- rotation (the coordinate origin is assumed to be the top-left corner).
  306     -> Double -- ^ Isotropic scale factor.
  307     -> Mat (ShapeT [2, 3]) ('S 1) ('S Double) -- ^ The output affine transformation, 2x3 floating-point matrix.
  308 getRotationMatrix2D center angle scale = unsafeCoerceMat $ unsafePerformIO $
  309     withPtr (toPoint center) $ \centerPtr ->
  310       fromPtr
  311       [CU.block| Mat * {
  312         return new cv::Mat
  313         ( cv::getRotationMatrix2D
  314           ( *$(Point2f * centerPtr)
  315           , $(double c'angle)
  316           , $(double c'scale)
  317           )
  318         );
  319       }|]
  320   where
  321     c'angle = realToFrac angle
  322     c'scale = realToFrac scale

  324 {- | Applies a generic geometrical transformation to an image.
  325 
  326 The function remap transforms the source image using the specified map:
  327 
  328 @dst(x,y) = src(map(x,y))@
  329 
  330 Example:
  331 
  332 @
  333 remapImg
  334   :: forall (width    :: Nat)
  335             (height   :: Nat)
  336             (channels :: Nat)
  337             (depth    :: *  )
  338    . (Mat ('S ['S height, 'S width]) ('S channels) ('S depth) ~ Birds_512x341)
  339   => Mat ('S ['S height, 'S width]) ('S channels) ('S depth)
  340 remapImg = exceptError $ remap birds_512x341 transform InterLinear (BorderConstant black)
  341   where
  342     transform = exceptError $
  343                 matFromFunc (Proxy :: Proxy [height, width])
  344                             (Proxy :: Proxy 2)
  345                             (Proxy :: Proxy Float)
  346                             exampleFunc
  347 
  348     exampleFunc [_y,  x] 0 = wobble x w
  349     exampleFunc [ y, _x] 1 = wobble y h
  350     exampleFunc _pos _channel = error "impossible"
  351 
  352     wobble :: Int -> Float -> Float
  353     wobble v s = let v' = fromIntegral v
  354                      n = v' / s
  355                  in v' + (s * 0.05 * sin (n * 2 * pi * 5))
  356 
  357     w = fromInteger $ natVal (Proxy :: Proxy width)
  358     h = fromInteger $ natVal (Proxy :: Proxy height)
  359 @
  360 
  361 <<doc/generated/birds_512x341.png original>>
  362 <<doc/generated/examples/remapImg.png remapImg>>
  363 
  364 <http://docs.opencv.org/3.0-last-rst/modules/imgproc/doc/geometric_transformations.html#remap OpenCV documentation>
  365 -}
  366 remap
  367     :: Mat ('S [inputHeight, inputWidth]) inputChannels inputDepth
  368        -- ^ Source image.
  369     -> Mat ('S [outputHeight, outputWidth]) ('S 2) ('S Float)
  370        -- ^ A map of @(x, y)@ points.
  371     -> InterpolationMethod
  372        -- ^ Interpolation method to use. Note that 'InterArea' is not
  373        -- supported by this function.
  374     -> BorderMode
  375     -> CvExcept (Mat ('S [outputHeight, outputWidth]) inputChannels inputDepth)
  376 remap src mapping interpolationMethod borderMode = unsafeWrapException $ do
  377     dst <- newEmptyMat
  378     handleCvException (pure $ unsafeCoerceMat dst) $
  379       withPtr src $ \srcPtr ->
  380       withPtr dst $ \dstPtr ->
  381       withPtr mapping $ \mappingPtr ->
  382       withPtr borderValue $ \borderValuePtr ->
  383         [cvExcept|
  384           cv::remap
  385             ( *$(Mat * srcPtr)
  386             , *$(Mat * dstPtr)
  387             , *$(Mat * mappingPtr)
  388             , {}
  389             , $(int32_t c'interpolation)
  390             , $(int32_t c'borderMode)
  391             , *$(Scalar * borderValuePtr)
  392             );
  393         |]
  394   where
  395     c'interpolation = marshalInterpolationMethod interpolationMethod
  396     (c'borderMode, borderValue) = marshalBorderMode borderMode
