  126 data OrbParams
  127    = OrbParams
  128      { orb_nfeatures :: !Int32
  129        -- ^ The maximum number of features to retain.
  130      , orb_scaleFactor :: !Float
  131        -- ^ Pyramid decimation ratio, greater than 1. 'orb_scaleFactor' == 2
  132        -- means the classical pyramid, where each next level has 4x less pixels
  133        -- than the previous, but such a big scale factor will degrade feature
  134        -- matching scores dramatically. On the other hand, too close to 1 scale
  135        -- factor will mean that to cover certain scale range you will need more
  136        -- pyramid levels and so the speed will suffer.
  137      , orb_nlevels :: !Int32
  138        -- ^ The number of pyramid levels. The smallest level will have linear
  139        -- size equal to input_image_linear_size / 'orb_scaleFactor' **
  140        -- 'orb_nlevels'.
  141      , orb_edgeThreshold :: !Int32
  142        -- ^ This is size of the border where the features are not detected. It
  143        -- should roughly match the patchSize parameter.
  144      , orb_firstLevel :: !Int32
  145        -- ^ It should be 0 in the current implementation.
  146      , orb_WTA_K :: !WTA_K
  147        -- ^ The number of points that produce each element of the oriented BRIEF
  148        -- descriptor. The default value 'WTA_K_2' means the BRIEF where we take
  149        -- a random point pair and compare their brightnesses, so we get 0/1
  150        -- response. Other possible values are 'WTA_K_3' and 'WTA_K_4'. For
  151        -- example, 'WTA_K_3' means that we take 3 random points (of course,
  152        -- those point coordinates are random, but they are generated from the
  153        -- pre-defined seed, so each element of BRIEF descriptor is computed
  154        -- deterministically from the pixel rectangle), find point of maximum
  155        -- brightness and output index of the winner (0, 1 or 2). Such output
  156        -- will occupy 2 bits, and therefore it will need a special variant of
  157        -- Hamming distance, denoted as 'Norm_Hamming2' (2 bits per bin). When
  158        -- 'WTA_K_4', we take 4 random points to compute each bin (that will also
  159        -- occupy 2 bits with possible values 0, 1, 2 or 3).
  160      , orb_scoreType :: !OrbScoreType
  161        -- ^ The default 'HarrisScore' means that Harris algorithm is used to
  162        -- rank features (the score is written to KeyPoint::score and is used to
  163        -- retain best nfeatures features); 'FastScore' is alternative value of
  164        -- the parameter that produces slightly less stable keypoints, but it is
  165        -- a little faster to compute.
  166      , orb_patchSize :: !Int32
  167        -- ^ Size of the patch used by the oriented BRIEF descriptor. Of course,
  168        -- on smaller pyramid layers the perceived image area covered by a
  169        -- feature will be larger.
  170      , orb_fastThreshold :: !Int32
  171      }
  172 
  173 defaultOrbParams :: OrbParams
  174 defaultOrbParams =

  176      { orb_nfeatures     = 500
  177      , orb_scaleFactor   = 1.2
  178      , orb_nlevels       = 8
  179      , orb_edgeThreshold = 31
  180      , orb_firstLevel    = 0
  181      , orb_WTA_K         = WTA_K_2
  182      , orb_scoreType     = HarrisScore
  183      , orb_patchSize     = 31
  184      , orb_fastThreshold = 20
  185      }

  216 {- | Detect keypoints and compute descriptors
  217 
  218 Example:
  219 
  220 @
  221 orbDetectAndComputeImg
  222     :: forall (width    :: Nat)
  223               (height   :: Nat)
  224               (channels :: Nat)
  225               (depth    :: *)
  226      . (Mat (ShapeT [height, width]) ('S channels) ('S depth) ~ Frog)
  227     => Mat (ShapeT [height, width]) ('S channels) ('S depth)
  228 orbDetectAndComputeImg = exceptError $ do
  229     (kpts, _descs) <- orbDetectAndCompute orb frog Nothing
  230     withMatM (Proxy :: Proxy [height, width])
  231              (Proxy :: Proxy channels)
  232              (Proxy :: Proxy depth)
  233              white $ \imgM -> do
  234       void $ matCopyToM imgM (V2 0 0) frog Nothing
  235       forM_ kpts $ \kpt -> do
  236         let kptRec = keyPointAsRec kpt
  237         circle imgM (round \<$> kptPoint kptRec :: V2 Int32) 5 blue 1 LineType_AA 0
  238   where
  239     orb = mkOrb defaultOrbParams
  240 @
  241 
  242 <<doc/generated/examples/orbDetectAndComputeImg.png orbDetectAndComputeImg>>
  243 -}
  244 orbDetectAndCompute
  245     :: Orb
  246     -> Mat ('S [height, width]) channels depth -- ^ Image.
  247     -> Maybe (Mat ('S [height, width]) ('S 1) ('S Word8)) -- ^ Mask.
  248     -> CvExcept ( V.Vector KeyPoint
  249                 , Mat 'D 'D 'D
  250                 )
  251 orbDetectAndCompute orb img mbMask = unsafeWrapException $ do
  252     descriptors <- newEmptyMat
  253     withPtr orb $ \orbPtr ->
  254       withPtr img $ \imgPtr ->
  255       withPtr mbMask $ \maskPtr ->
  256       withPtr descriptors $ \descPtr ->
  257       alloca $ \(numPtsPtr :: Ptr Int32) ->
  258       alloca $ \(arrayPtrPtr :: Ptr (Ptr (Ptr C'KeyPoint))) -> mask_ $ do
  259         ptrException <- [cvExcept|
  260           cv::ORB * orb = *$(Ptr_ORB * orbPtr);
  261           cv::Mat * maskPtr = $(Mat * maskPtr);
  262 
  263           std::vector<cv::KeyPoint> keypoints = std::vector<cv::KeyPoint>();
  264           orb->
  265             detectAndCompute
  266             ( *$(Mat * imgPtr)
  267             , maskPtr ? cv::_InputArray(*maskPtr) : cv::_InputArray(noArray())
  268             , keypoints
  269             , *$(Mat * descPtr)
  270             , false
  271             );
  272 
  273           *$(int32_t * numPtsPtr) = keypoints.size();
  274 
  275           cv::KeyPoint * * * arrayPtrPtr = $(KeyPoint * * * arrayPtrPtr);
  276           cv::KeyPoint * * arrayPtr = new cv::KeyPoint * [keypoints.size()];
  277           *arrayPtrPtr = arrayPtr;
  278 
  279           for (std::vector<cv::KeyPoint>::size_type ix = 0; ix != keypoints.size(); ix++)
  280           {
  281             cv::KeyPoint & org = keypoints[ix];
  282             cv::KeyPoint * newPt =
  283               new cv::KeyPoint( org.pt
  284                               , org.size
  285                               , org.angle
  286                               , org.response
  287                               , org.octave
  288                               , org.class_id
  289                               );
  290             arrayPtr[ix] = newPt;
  291           }
  292         |]
  293         if ptrException /= nullPtr
  294         then Left . BindingException <$> fromPtr (pure ptrException)
  295         else do
  296           numPts <- fromIntegral <$> peek numPtsPtr
  297           arrayPtr <- peek arrayPtrPtr
  298           keypoints <- mapM (fromPtr . pure) =<< peekArray numPts arrayPtr
  299 
  300           [CU.block| void {
  301             delete [] *$(KeyPoint * * * arrayPtrPtr);
  302           }|]
  303 
  304           pure $ Right (V.fromList keypoints, relaxMat descriptors)

  455 blobDetect
  456     :: SimpleBlobDetector
  457     -> Mat ('S [height, width]) channels depth -- ^ Image.
  458     -> Maybe (Mat ('S [height, width]) ('S 1) ('S Word8)) -- ^ Mask.
  459     -> CvExcept (V.Vector KeyPoint)
  460 blobDetect detector img mbMask = unsafeWrapException $ do
  461     withPtr detector $ \detectorPtr ->
  462       withPtr img $ \imgPtr ->
  463       withPtr mbMask $ \maskPtr ->
  464       alloca $ \(numPtsPtr :: Ptr Int32) ->
  465       alloca $ \(arrayPtrPtr :: Ptr (Ptr (Ptr C'KeyPoint))) -> mask_ $ do
  466         ptrException <- [cvExcept|
  467           cv::SimpleBlobDetector * detector = *$(Ptr_SimpleBlobDetector * detectorPtr);
  468           cv::Mat * maskPtr = $(Mat * maskPtr);
  469 
  470           std::vector<cv::KeyPoint> keypoints = std::vector<cv::KeyPoint>();
  471           detector->
  472             detect
  473             ( *$(Mat * imgPtr)
  474             , keypoints
  475             , maskPtr ? cv::_InputArray(*maskPtr) : cv::_InputArray(noArray())
  476             );
  477 
  478           *$(int32_t * numPtsPtr) = keypoints.size();
  479 
  480           cv::KeyPoint * * * arrayPtrPtr = $(KeyPoint * * * arrayPtrPtr);
  481           cv::KeyPoint * * arrayPtr = new cv::KeyPoint * [keypoints.size()];
  482           *arrayPtrPtr = arrayPtr;
  483 
  484           for (std::vector<cv::KeyPoint>::size_type ix = 0; ix != keypoints.size(); ix++)
  485           {
  486             arrayPtr[ix] = new cv::KeyPoint(keypoints[ix]);
  487           }
  488         |]
  489         if ptrException /= nullPtr
  490         then Left . BindingException <$> fromPtr (pure ptrException)
  491         else do
  492           numPts <- fromIntegral <$> peek numPtsPtr
  493           arrayPtr <- peek arrayPtrPtr
  494           keypoints <- mapM (fromPtr . pure) =<< peekArray numPts arrayPtr
  495 
  496           [CU.block| void {
  497             delete [] *$(KeyPoint * * * arrayPtrPtr);
  498           }|]
  499 
  500           pure $ Right (V.fromList keypoints)

  506 class DescriptorMatcher a where
  507     match
  508         :: a
  509         -> Mat 'D 'D 'D -- ^ Query set of descriptors.
  510         -> Mat 'D 'D 'D -- ^ Train set of descriptors.
  511         -> Maybe (Mat ('S [height, width]) ('S 1) ('S Word8))
  512            -- ^ Mask specifying permissible matches between an input query and
  513            -- train matrices of descriptors..
  514         -> IO (V.Vector DMatch)

  516 --------------------------------------------------------------------------------
  517 -- BFMatcher
  518 --------------------------------------------------------------------------------
  519 
  520 {- | Brute-force descriptor matcher
  521 
  522 For each descriptor in the first set, this matcher finds the closest descriptor
  523 in the second set by trying each one. This descriptor matcher supports masking
  524 permissible matches of descriptor sets.
  525 
  526 Example:
  527 
  528 @
  529 bfMatcherImg
  530     :: forall (width    :: Nat)
  531               (width2   :: Nat)
  532               (height   :: Nat)
  533               (channels :: Nat)
  534               (depth    :: *)
  535      . ( Mat (ShapeT [height, width]) ('S channels) ('S depth) ~ Frog
  536        , width2 ~ (*) width 2
  537        )
  538     => IO (Mat (ShapeT [height, width2]) ('S channels) ('S depth))
  539 bfMatcherImg = do
  540     let (kpts1, descs1) = exceptError $ orbDetectAndCompute orb frog        Nothing
  541         (kpts2, descs2) = exceptError $ orbDetectAndCompute orb rotatedFrog Nothing
  542 
  543     bfmatcher <- newBFMatcher Norm_Hamming True
  544     matches <- match bfmatcher
  545                      descs1 -- Query descriptors
  546                      descs2 -- Train descriptors
  547                      Nothing
  548     exceptErrorIO $ pureExcept $
  549       withMatM (Proxy :: Proxy [height, width2])
  550                (Proxy :: Proxy channels)
  551                (Proxy :: Proxy depth)
  552                white $ \imgM -> do
  553         matCopyToM imgM (V2 0     0) frog        Nothing
  554         matCopyToM imgM (V2 width 0) rotatedFrog Nothing
  555 
  556         -- Draw the matches as lines from the query image to the train image.
  557         forM_ matches $ \dmatch -> do
  558           let matchRec = dmatchAsRec dmatch
  559               queryPt = kpts1 V.! fromIntegral (dmatchQueryIdx matchRec)
  560               trainPt = kpts2 V.! fromIntegral (dmatchTrainIdx matchRec)
  561               queryPtRec = keyPointAsRec queryPt
  562               trainPtRec = keyPointAsRec trainPt
  563 
  564           -- We translate the train point one width to the right in order to
  565           -- match the position of rotatedFrog in imgM.
  566           line imgM
  567                (round \<$> kptPoint queryPtRec :: V2 Int32)
  568                ((round \<$> kptPoint trainPtRec :: V2 Int32) ^+^ V2 width 0)
  569                blue 1 LineType_AA 0
  570   where

  597 newBFMatcher
  598     :: NormType
  599        -- ^ 'Norm_L1' and 'Norm_L2' norms are preferable choices for SIFT and
  600        -- SURF descriptors, 'Norm_Hamming' should be used with 'Orb', BRISK and
  601        -- BRIEF, 'Norm_Hamming2' should be used with 'Orb' when 'WTA_K_3' or
  602        -- 'WTA_K_4' (see 'orb_WTA_K').
  603     -> Bool
  604        -- ^ If it is false, this is will be default 'BFMatcher' behaviour when
  605        -- it finds the k nearest neighbors for each query descriptor. If
  606        -- crossCheck == True, then the @knnMatch()@ method with @k=1@ will only
  607        -- return pairs @(i,j)@ such that for i-th query descriptor the j-th
  608        -- descriptor in the matcher's collection is the nearest and vice versa,
  609        -- i.e. the 'BFMatcher' will only return consistent pairs. Such technique
  610        -- usually produces best results with minimal number of outliers when
  611        -- there are enough matches. This is alternative to the ratio test, used
  612        -- by D. Lowe in SIFT paper.
  613     -> IO BFMatcher
  614 newBFMatcher normType crossCheck = fromPtr
  615     [CU.exp|BFMatcher * {
  616       new cv::BFMatcher
  617           ( $(int32_t c'normType)
  618           , $(bool c'crossCheck)
  619           )
  620     }|]
  621   where
  622     c'normType = marshalNormType NormAbsolute normType
  623     c'crossCheck = fromBool crossCheck

  627 instance DescriptorMatcher BFMatcher where
  628   match bf queryDescriptors trainDescriptors mbMask =
  629       withPtr bf               $ \bfPtr    ->
  630       withPtr queryDescriptors $ \queryPtr ->
  631       withPtr trainDescriptors $ \trainPtr ->
  632       withPtr mbMask           $ \maskPtr  ->
  633       alloca $ \(numMatchesPtr :: Ptr Int32) ->
  634       alloca $ \(arrayPtrPtr :: Ptr (Ptr (Ptr C'DMatch))) -> mask_ $ do
  635         [C.block| void {
  636           cv::Mat * maskPtr = $(Mat * maskPtr);
  637           std::vector<cv::DMatch> matches = std::vector<cv::DMatch>();
  638 
  639           $(BFMatcher * bfPtr)->match
  640             ( *$(Mat * queryPtr)
  641             , *$(Mat * trainPtr)
  642             , matches
  643             , maskPtr ? cv::_InputArray(*maskPtr) : cv::_InputArray(noArray())
  644             );
  645 
  646           *$(int32_t * numMatchesPtr) = matches.size();
  647 
  648           cv::DMatch * * * arrayPtrPtr = $(DMatch * * * arrayPtrPtr);
  649           cv::DMatch * * arrayPtr = new cv::DMatch * [matches.size()];
  650           *arrayPtrPtr = arrayPtr;
  651 
  652           for (std::vector<cv::DMatch>::size_type ix = 0; ix != matches.size(); ix++)
  653           {
  654             cv::DMatch & org = matches[ix];
  655             cv::DMatch * newMatch =
  656               new cv::DMatch( org.queryIdx
  657                             , org.trainIdx
  658                             , org.imgIdx
  659                             , org.distance
  660                             );
  661             arrayPtr[ix] = newMatch;
  662           }
  663         }|]
  664 
  665         (numMatches :: Int) <- fromIntegral <$> peek numMatchesPtr
  666         arrayPtr <- peek arrayPtrPtr
  667         matches <- mapM (fromPtr . pure) =<< peekArray numMatches arrayPtr
  668 
  669         [CU.block| void {
  670           delete [] *$(DMatch * * * arrayPtrPtr);
  671         }|]
  672 
  673         pure $ V.fromList matches
