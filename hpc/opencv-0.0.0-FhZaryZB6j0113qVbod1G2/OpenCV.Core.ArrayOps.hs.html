   98 matScalarAdd
   99     :: (ToScalar scalar)
  100     => Mat shape channels depth -- ^
  101     -> scalar
  102     -> Mat shape channels depth
  103 matScalarAdd src x = unsafePerformIO $ do
  104     dst <- newEmptyMat
  105     withPtr (toScalar x) $ \xPtr ->
  106       withPtr dst $ \dstPtr ->
  107         withPtr src $ \srcPtr ->
  108           [C.block| void {
  109             *$(Mat * dstPtr) = *$(Mat * srcPtr) + *$(Scalar * xPtr);
  110           }|]
  111     pure $ unsafeCoerceMat dst
  112 
  113 matScalarMult
  114     :: Mat shape channels depth -- ^
  115     -> Double
  116     -> Mat shape channels depth
  117 matScalarMult src x = unsafePerformIO $ do
  118     dst <- newEmptyMat
  119     withPtr dst $ \dstPtr ->
  120       withPtr src $ \srcPtr ->
  121         [C.block| void {
  122           *$(Mat * dstPtr) = *$(Mat * srcPtr) * $(double c'x);
  123         }|]
  124     pure $ unsafeCoerceMat dst
  125   where
  126     c'x = realToFrac x

  304 matScaleAdd src1 scale src2 = unsafeWrapException $ do
  305     dst <- newEmptyMat

  307       withPtr src1 $ \src1Ptr ->
  308       withPtr src2 $ \src2Ptr ->
  309       withPtr dst  $ \dstPtr  ->
  310       [cvExcept|
  311         cv::scaleAdd
  312           ( *$(Mat * src1Ptr)
  313           , $(double c'scale)
  314           , *$(Mat * src2Ptr)
  315           , *$(Mat * dstPtr)
  316           );
  317       |]
  318   where
  319     c'scale = realToFrac scale

  569 {- | Divides a multi-channel array into several single-channel arrays.
  570 
  571 Example:
  572 
  573 @
  574 matSplitImg
  575     :: forall (width    :: Nat)
  576               (width3   :: Nat)
  577               (height   :: Nat)
  578               (channels :: Nat)
  579               (depth    :: *)
  580      . ( Mat (ShapeT [height, width]) ('S channels) ('S depth) ~ Birds_512x341
  581        , width3 ~ ((*) width 3)
  582        )
  583     => Mat (ShapeT [height, width3]) ('S channels) ('S depth)
  584 matSplitImg = exceptError $ do
  585     zeroImg <- mkMat (Proxy :: Proxy [height, width])
  586                      (Proxy :: Proxy 1)
  587                      (Proxy :: Proxy depth)
  588                      black
  589     let blueImg  = matMerge $ V.fromList [channelImgs V.! 0, zeroImg, zeroImg]
  590         greenImg = matMerge $ V.fromList [zeroImg, channelImgs V.! 1, zeroImg]
  591         redImg   = matMerge $ V.fromList [zeroImg, zeroImg, channelImgs V.! 2]
  592 
  593     withMatM (Proxy :: Proxy [height, width3])
  594              (Proxy :: Proxy channels)
  595              (Proxy :: Proxy depth)
  596              white $ \imgM -> do
  597       matCopyToM imgM (V2 (w*0) 0) (unsafeCoerceMat blueImg)  Nothing
  598       matCopyToM imgM (V2 (w*1) 0) (unsafeCoerceMat greenImg) Nothing
  599       matCopyToM imgM (V2 (w*2) 0) (unsafeCoerceMat redImg)   Nothing
  600   where
  601     channelImgs = matSplit birds_512x341
  602 
  603     w :: Int32
  604     w = fromInteger $ natVal (Proxy :: Proxy width)
  605 @
  606 
  607 <<doc/generated/examples/matSplitImg.png matSplitImg>>
  608 
  609 <http://docs.opencv.org/3.0-last-rst/modules/core/doc/operations_on_arrays.html#split OpenCV Sphinx doc>
  610 -}
  611 matSplit
  612     :: Mat shape channels depth -- ^
  613     -> V.Vector (Mat shape ('S 1) depth)
  614 matSplit src = unsafePerformIO $
  615     withPtr src $ \srcPtr ->
  616     allocaArray numChans $ \(splitsArray :: Ptr (Ptr C'Mat)) -> do
  617       [C.block| void {
  618         cv::Mat * srcPtr = $(Mat * srcPtr);
  619         int32_t numChans = $(int32_t c'numChans);
  620         cv::Mat *splits = new cv::Mat[numChans];
  621         cv::split(*srcPtr, splits);
  622         for(int i = 0; i < numChans; i++){
  623           $(Mat * * splitsArray)[i] = new cv::Mat(splits[i]);
  624         }
  625         delete [] splits;
  626       }|]
  627       fmap V.fromList . mapM (fromPtr . pure) =<< peekArray numChans splitsArray
  628   where
  629     numChans = fromIntegral $ miChannels $ matInfo src
  630 
  631     c'numChans :: Int32
  632     c'numChans = fromIntegral numChans

  658       alloca $ \maxValPtr -> do

  660           ( (,, minLoc, maxLoc)
  661             <$> (realToFrac <$> peek minValPtr)
  662             <*> (realToFrac <$> peek maxValPtr)
  663           )
  664           [cvExcept|
  665             cv::minMaxLoc( *$(Mat * srcPtr)
  666                          , $(double * minValPtr)
  667                          , $(double * maxValPtr)
  668                          , $(Point2i * minLocPtr)
  669                          , $(Point2i * maxLocPtr)
  670                          );
  671           |]

  731 {- | Normalizes the norm or value range of an array
  732 
  733 <http://docs.opencv.org/3.0-last-rst/modules/core/doc/operations_on_arrays.html#normalize OpenCV Sphinx doc>
  734 -}
  735 normalize
  736     :: forall shape channels srcDepth dstDepth
  737      . (ToDepthDS (Proxy dstDepth))
  738     => Double
  739        -- ^ Norm value to normalize to or the lower range boundary in case of
  740        -- the range normalization.
  741     -> Double
  742        -- ^ Upper range boundary in case of the range normalization; it is not
  743        -- used for the norm normalization.
  744     -> NormType
  745     -> Maybe (Mat shape ('S 1) ('S Word8)) -- ^ Optional operation mask.
  746     -> Mat shape channels srcDepth -- ^ Input array.
  747     -> CvExcept (Mat shape channels dstDepth)
  748 normalize alpha beta normType mbMask src = unsafeWrapException $ do
  749     dst <- newEmptyMat
  750     handleCvException (pure $ unsafeCoerceMat dst) $
  751       withPtr src    $ \srcPtr ->
  752       withPtr dst    $ \dstPtr ->
  753       withPtr mbMask $ \mskPtr ->
  754         [cvExcept|
  755           Mat * mskPtr = $(Mat * mskPtr);
  756           cv::normalize( *$(Mat * srcPtr)
  757                        , *$(Mat * dstPtr)
  758                        , $(double c'alpha)
  759                        , $(double c'beta)
  760                        , $(int32_t c'normType)
  761                        , $(int32_t c'dtype)
  762                        , mskPtr ? _InputArray(*mskPtr) : _InputArray(cv::noArray())
  763                        );
  764         |]
  765   where
  766     c'alpha    = realToFrac alpha
  767     c'beta     = realToFrac beta
  768     c'normType = marshalNormType NormAbsolute normType
  769     c'dtype    = maybe (-1) marshalDepth $ dsToMaybe $ toDepthDS (Proxy :: Proxy dstDepth)

  855 {- | Flips a 2D matrix around vertical, horizontal, or both axes.
  856 
  857 The example scenarios of using the function are the following: Vertical flipping
  858 of the image ('FlipVertically') to switch between top-left and bottom-left image
  859 origin. This is a typical operation in video processing on Microsoft Windows*
  860 OS. Horizontal flipping of the image with the subsequent horizontal shift and
  861 absolute difference calculation to check for a vertical-axis symmetry
  862 ('FlipHorizontally'). Simultaneous horizontal and vertical flipping of the image
  863 with the subsequent shift and absolute difference calculation to check for a
  864 central symmetry ('FlipBoth'). Reversing the order of point arrays
  865 ('FlipHorizontally' or 'FlipVertically').
  866 
  867 Example:
  868 
  869 @
  870 matFlipImg :: Mat (ShapeT [341, 512]) ('S 3) ('S Word8)
  871 matFlipImg = matFlip sailboat_512x341 FlipBoth
  872 @
  873 
  874 <<doc/generated/examples/matFlipImg.png matFlipImg>>
  875 -}
  876 matFlip
  877     :: Mat ('S '[height, width]) channels depth
  878     -> FlipDirection -- ^ How to flip.
  879     -> Mat ('S '[height, width]) channels depth
  880 matFlip src flipDir = unsafePerformIO $ do
  881     dst <- newEmptyMat
  882     withPtr dst $ \dstPtr ->
  883       withPtr src $ \srcPtr ->
  884         [C.block| void {
  885           cv::flip(*$(Mat * srcPtr), *$(Mat * dstPtr), $(int32_t flipCode));
  886         }|]
  887     pure $ unsafeCoerceMat dst
  888   where
  889     flipCode :: Int32
  890     flipCode = marshallFlipDirection flipDir
  891 
  892 data FlipDirection = FlipVertically   -- ^ Flip around the x-axis.
  893                    | FlipHorizontally -- ^ Flip around the y-axis.
  894                    | FlipBoth         -- ^ Flip around both x and y-axis.
  895                      deriving (Show, Eq)
  896 
  897 marshallFlipDirection :: FlipDirection -> Int32
  898 marshallFlipDirection = \case
  899     FlipVertically   ->  0
  900     FlipHorizontally ->  1
  901     FlipBoth         -> -1
