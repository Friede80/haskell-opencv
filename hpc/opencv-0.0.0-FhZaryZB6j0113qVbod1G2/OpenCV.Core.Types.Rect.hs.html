   41 fmapRect :: forall a b. (IsRect Rect a, IsRect HRect a, IsRect Rect b, IsRect HRect b)
   42    => (a -> b) -> Rect a -> Rect b
   43 fmapRect f r = toRect hrB
   44   where
   45     hrA :: HRect a
   46     hrA = fromRect r
   47 
   48     hrB :: HRect b
   49     hrB = fmap f hrA
